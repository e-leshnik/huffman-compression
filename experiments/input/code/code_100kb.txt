class AvlTree<K : Comparable<K>, V> : Tree<K, V> {

    }

    private fun updateHeight(node: AvlNode<K, V>?) {
        node?.height = 1 + maxOf(height(node.left), height(node.right))
    }

    private fun balanceFactor(node: AvlNode<K, V>?): Int {
        return if (node == null) 0 else height(node.left) - height(node.right)
    }

    private fun minNode(node: AvlNode<K, V>): AvlNode<K, V> {
        var current: AvlNode<K, V> = node
        while (current.left != null) {
            current = current.left!!
        }
        return current
    }

enum class NodeColor {
    RED, BLACK
}

class RBNode<K : Comparable<K>, V>(
    key: K,
    value: V,
    left: RBNode<K, V>? = null,
    right: RBNode<K, V>? = null,
    var parent: RBNode<K, V>? = null,
    var color: NodeColor = NodeColor.RED,
) : Node<K, V, RBNode<K, V>>(key, value, left, right)

class RBTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: RBNode<K, V>? = null

    private fun rotateLeft(x: RBNode<K, V>?) {         //x        y
        val y = x?.right ?: return                     // \  ->  /
                                                       //  y    x
        x.right = y.left
        y.left?.parent = x

        val parent = x.parent
        y.parent = parent

        when {
            parent == null -> root = y
            x == parent.left -> parent.left = y
            else -> parent.right = y
        }

        y.left = x
        x.parent = y
    }

    private fun rotateRight(x: RBNode<K, V>?) {        // x     y
        val y = x?.left ?: return                     // /  ->   \
                                                     // y         x
        x.left = y.right
        y.right?.parent = x

        val parent = x.parent
        y.parent = parent

        when {
            parent == null -> root = y
            x == parent.right -> parent.right = y
            else -> parent.left = y
        }

        y.right = x
        x.parent = y
    }

    private fun fixAfterInsert(node: RBNode<K, V>) {
        var current: RBNode<K, V>? = node

        while (current != root && current?.parent?.color == NodeColor.RED) {
            val grandparent = current.parent?.parent

            if (current.parent == grandparent?.left) {
                val uncle = grandparent?.right

                if (uncle?.color == NodeColor.RED) {
                    current.parent?.color = NodeColor.BLACK
                    uncle.color = NodeColor.BLACK
                    grandparent.color = NodeColor.RED
                    current = grandparent
                } else {
                    if (current == current.parent?.right) {
                        val parent = current.parent

                        if (parent != null) {
                            current = parent
                            rotateLeft(current)
                        }
                    }

                    current.parent?.color = NodeColor.BLACK
                    grandparent?.color = NodeColor.RED
                    rotateRight(grandparent)
                }
            } else {
                val uncle = grandparent?.left

                if (uncle?.color == NodeColor.RED) {
                    current.parent?.color = NodeColor.BLACK
                    uncle.color = NodeColor.BLACK
                    grandparent.color = NodeColor.RED
                    current = grandparent
                } else {
                    if (current == current.parent?.left) {
                        current = current.parent
                        rotateRight(current)
                    }

                    current?.parent?.color = NodeColor.BLACK
                    grandparent?.color = NodeColor.RED
                    rotateLeft(grandparent)
                }
            }
        }

        root?.color = NodeColor.BLACK
    }

class BinTree<K: Comparable<K>, V>: Tree<K, V> {
    private var root: BinNode<K, V>? = null

    override fun insert(key: K, value: V) {
        if (root == null) {
            root = BinNode(key, value)
            return
        }

        var current = root

        while (current != null) {
            when {
                key < current.key -> {
                    if (current.left == null) {
                        current.left = BinNode(key, value)
                        return
                    }
                    current = current.left
                }

                key > current.key -> {
                    if (current.right == null) {
                        current.right = BinNode(key, value)
                        return
                    }
                    current = current.right
                }

                else -> {
                    current.value = value
                    return
                }
            }
        }
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(binNode: BinNode<K, V>?, key: K): BinNode<K, V>? {
        binNode ?: return null

        when {
            key < binNode.key -> binNode.left = deleteNode(binNode.left, key)
            key > binNode.key -> binNode.right = deleteNode(binNode.right, key)
            else -> {
                if (binNode.left == null && binNode.right == null) {
                    return null
                }
                if (binNode.left == null) {
                    return binNode.right
                }
                if (binNode.right == null) {
                    return binNode.left
                }

                val minNode = findMin(binNode.right!!)
                binNode.key = minNode.key
                binNode.value = minNode.value
                binNode.right = deleteNode(binNode.right, minNode.key)
            }
        }
        return binNode
    }

    private fun findMin(binNode: BinNode<K, V>): BinNode<K, V> {
        var current = binNode
        while (current.left != null) {
            current = current.left!!
        }
        return current
    }

    override fun search(key: K): V? {
        var current = root

        while (current != null) {
            when {
                key < current.key -> current = current.left
                key > current.key -> current = current.right
                else -> return current.value
            }
        }

        return null
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        val result = mutableListOf<Pair<K, V>>()
        val queue = ArrayDeque<BinNode<K, V>>()

        if (root != null) {
            queue.add(root!!)
        }

        while (queue.isNotEmpty()) {
            val current = queue.removeFirst()
            result.add(current.key to current.value)

            if (current.left != null) {
                queue.add(current.left!!)
            }
            if (current.right != null) {
                queue.add(current.right!!)
            }
        }

        return result.iterator()
    }

}class AvlNode<K : Comparable<K>, V>(
    key: K,
    value: V,
    left: AvlNode<K, V>? = null,
    right: AvlNode<K, V>? = null,
    var height: Int = 1
) : Node<K, V, AvlNode<K, V>>(key, value, left, right)

class AvlTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: AvlNode<K, V>? = null
    var size = 0

    private fun height(node: AvlNode<K, V>?): Int {
        return node?.height ?: 0
    }

    private fun updateHeight(node: AvlNode<K, V>?) {
        node?.height = 1 + maxOf(height(node.left), height(node.right))
    }

    private fun balanceFactor(node: AvlNode<K, V>?): Int {
        return if (node == null) 0 else height(node.left) - height(node.right)
    }

    private fun minNode(node: AvlNode<K, V>): AvlNode<K, V> {
        var current: AvlNode<K, V> = node
        while (true) {
            val left = current.left ?: return current
            current = left
        }
    }

    private fun rotateRight(node: AvlNode<K, V>): AvlNode<K, V> {
        val newRoot: AvlNode<K, V> = node.left ?: return node
        val temp = newRoot.right
        newRoot.right = node
        node.left = temp
        updateHeight(node)
        updateHeight(newRoot)
        return newRoot
    }

    private fun rotateLeft(node: AvlNode<K, V>): AvlNode<K, V> {
        val newRoot : AvlNode<K, V> = node.right ?: return node
        val temp = newRoot.left
        newRoot.left = node
        node.right = temp
        updateHeight(node)
        updateHeight(newRoot)
        return newRoot
    }

    private fun rotateRightLeft(node: AvlNode<K, V>): AvlNode<K, V> {
        node.right = node.right?.let { rotateRight(it) }
        return rotateLeft(node)
    }

    private fun rotateLeftRight(node: AvlNode<K, V>): AvlNode<K, V> {
        node.left = node.left?.let { rotateLeft(it) }
        return rotateRight(node)
    }

    private fun balance(node: AvlNode<K, V>): AvlNode<K, V> {
        updateHeight(node)
        val balance = balanceFactor(node)

        if (balance > 1) {
            if (balanceFactor(node.left) >= 0) return rotateRight(node)
            return rotateLeftRight(node)

        }

        if (balance < -1) {
            if (balanceFactor(node.right) <= 0) return rotateLeft(node)
            return rotateRightLeft(node)
        }

        return node
    }

    override fun search(key: K): V? {
        val target = searchNode(root, key)
        if (target != null) return target.value
        return null
    }

    private fun searchNode(node: AvlNode<K, V>?, key: K): AvlNode<K, V>? {
        if (node == null || node.key == key) return node
        if (node.key < key) return searchNode(node.right, key)
        return searchNode(node.left, key)
    }

    override fun insert(key: K, value: V) {
        root = insertNode(root, key, value)
    }

    private fun insertNode(node: AvlNode<K, V>?, key: K, value: V): AvlNode<K, V> {
        if (node == null) {
            size++
            return AvlNode(key, value)
        }
        if (key < node.key) {
            node.left = insertNode(node.left, key, value)
        } else if (key > node.key) {
            node.right = insertNode(node.right, key, value)
        } else {
            node.value = value
            return node
        }
        return balance(node)
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(node: AvlNode<K, V>?, key: K): AvlNode<K, V>? {
        if (node == null) return null
        if (key < node.key) {
            node.left = deleteNode(node.left, key)
        } else if (key > node.key) {
            node.right = deleteNode(node.right, key)
        } else {
            if (node.left == null) {
                size--
                return node.right
            } else if (node.right == null) {
                size--
                return node.left
            }
            node.right?.let { right ->
                val temp = minNode(right)
                node.key = temp.key
                node.value = temp.value
                node.right = deleteNode(node.right, temp.key)
            } ?: error("Right  child should not be null")
        }
        return balance(node)
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<AvlNode<K, V>>()
        root?.let { queue.add(it) }
        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }
}

    value: V,
    left: AvlNode<K, V>? = null,
    right: AvlNode<K, V>? = null,
    var height: Int = 1
) : Node<K, V, AvlNode<K, V>>(key, value, left, right)

class AvlTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: AvlNode<K, V>? = null
    var size = 0

    private fun height(node: AvlNode<K, V>?): Int {
        return node?.height ?: 0
    }

    private fun updateHeight(node: AvlNode<K, V>?) {
        node?.height = 1 + maxOf(height(node.left), height(node.right))
    }

    private fun balanceFactor(node: AvlNode<K, V>?): Int {
        return if (node == null) 0 else height(node.left) - height(node.right)
    }

    private fun minNode(node: AvlNode<K, V>): AvlNode<K, V> {
        var current: AvlNode<K, V> = node
        while (true) {
            val left = current.left ?: return current
            current = left
        }
    }

    private fun rotateRight(node: AvlNode<K, V>): AvlNode<K, V> {
        val newRoot: AvlNode<K, V> = node.left ?: return node
        val temp = newRoot.right
        newRoot.right = node
        node.left = temp
        updateHeight(node)
        updateHeight(newRoot)
        return newRoot
    }

    private fun rotateLeft(node: AvlNode<K, V>): AvlNode<K, V> {
        val newRoot : AvlNode<K, V> = node.right ?: return node
        val temp = newRoot.left
        newRoot.left = node
        node.right = temp
        updateHeight(node)
        updateHeight(newRoot)
        return newRoot
    }

    private fun rotateRightLeft(node: AvlNode<K, V>): AvlNode<K, V> {
        node.right = node.right?.let { rotateRight(it) }
        return rotateLeft(node)
    }

    private fun rotateLeftRight(node: AvlNode<K, V>): AvlNode<K, V> {
        node.left = node.left?.let { rotateLeft(it) }
        return rotateRight(node)
    }

    private fun balance(node: AvlNode<K, V>): AvlNode<K, V> {
        updateHeight(node)
        val balance = balanceFactor(node)

        if (balance > 1) {
            if (balanceFactor(node.left) >= 0) return rotateRight(node)
            return rotateLeftRight(node)

        }

        if (balance < -1) {
            if (balanceFactor(node.right) <= 0) return rotateLeft(node)
            return rotateRightLeft(node)
        }

        return node
    }

    override fun search(key: K): V? {
        val target = searchNode(root, key)
        if (target != null) return target.value
        return null
    }

    private fun searchNode(node: AvlNode<K, V>?, key: K): AvlNode<K, V>? {
        if (node == null || node.key == key) return node
        if (node.key < key) return searchNode(node.right, key)
        return searchNode(node.left, key)
    }

    override fun insert(key: K, value: V) {
        root = insertNode(root, key, value)
    }

    private fun insertNode(node: AvlNode<K, V>?, key: K, value: V): AvlNode<K, V> {
        if (node == null) {
            size++
            return AvlNode(key, value)
        }
        if (key < node.key) {
            node.left = insertNode(node.left, key, value)
        } else if (key > node.key) {
            node.right = insertNode(node.right, key, value)
        } else {
            node.value = value
            return node
        }
        return balance(node)
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(node: AvlNode<K, V>?, key: K): AvlNode<K, V>? {
        if (node == null) return null
        if (key < node.key) {
            node.left = deleteNode(node.left, key)
        } else if (key > node.key) {
            node.right = deleteNode(node.right, key)
        } else {
            if (node.left == null) {
                size--
                return node.right
            } else if (node.right == null) {
                size--
                return node.left
            }
            node.right?.let { right ->
                val temp = minNode(right)
                node.key = temp.key
                node.value = temp.value
                node.right = deleteNode(node.right, temp.key)
            } ?: error("Right  child should not be null")
        }
        return balance(node)
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<AvlNode<K, V>>()
        root?.let { queue.add(it) }
        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }
}
class BinNode<K : Comparable<K>, V>(
    key: K,
    value: V,
    left: BinNode<K, V>? = null,
    right: BinNode<K, V>? = null
) : Node<K, V, BinNode<K, V>>(key, value, left, right)

class BinTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: BinNode<K, V>? = null

    override fun insert(key: K, value: V) {
        if (root == null) {
            root = BinNode(key, value)
            return
        }

        var current = root

        while (current != null) {
            when {
                key < current.key -> {
                    if (current.left == null) {
                        current.left = BinNode(key, value)
                        return
                    }
                    current = current.left
                }

                key > current.key -> {
                    if (current.right == null) {
                        current.right = BinNode(key, value)
                        return
                    }
                    current = current.right
                }

                else -> {
                    current.value = value
                    return
                }
            }
        }
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(node: BinNode<K, V>?, key: K): BinNode<K, V>? {
        node ?: return null

        when {
            key < node.key -> node.left = deleteNode(node.left, key)
            key > node.key -> node.right = deleteNode(node.right, key)
            else -> {
                if (node.left == null && node.right == null) {
                    return null
                }
                if (node.left == null) {
                    return node.right
                }
                if (node.right == null) {
                    return node.left
                }

                val rightChild = node.right ?: return node
                val minNode = findMin(rightChild)
                node.key = minNode.key
                node.value = minNode.value
                node.right = deleteNode(node.right, minNode.key)
            }
        }
        return node
    }

    private fun findMin(node: BinNode<K, V>): BinNode<K, V> {
        var current = node
        while (true) {
            val leftChild = current.left ?: break
            current = leftChild
        }
        return current
    }

    override fun search(key: K): V? {
        var current = root

        while (current != null) {
            when {
                key < current.key -> current = current.left
                key > current.key -> current = current.right
                else -> return current.value
            }
        }

        return null
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<BinNode<K, V>>()
        root?.let { queue.add(it) }

            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }
}class BinNode<K : Comparable<K>, V>(
    key: K,
    value: V,
    left: BinNode<K, V>? = null,
    right: BinNode<K, V>? = null
) : Node<K, V, BinNode<K, V>>(key, value, left, right)

class BinTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: BinNode<K, V>? = null

    override fun insert(key: K, value: V) {
        if (root == null) {
            root = BinNode(key, value)
            return
        }

        var current = root

        while (current != null) {
            when {
                key < current.key -> {
                    if (current.left == null) {
                        current.left = BinNode(key, value)
                        return
                    }
                    current = current.left
                }

                key > current.key -> {
                    if (current.right == null) {
                        current.right = BinNode(key, value)
                        return
                    }
                    current = current.right
                }

                else -> {
                    current.value = value
                    return
                }
            }
        }
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(node: BinNode<K, V>?, key: K): BinNode<K, V>? {
        node ?: return null

        when {
            key < node.key -> node.left = deleteNode(node.left, key)
            key > node.key -> node.right = deleteNode(node.right, key)
            else -> {
                if (node.left == null && node.right == null) {
                    return null
                }
                if (node.left == null) {
                    return node.right
                }
                if (node.right == null) {
                    return node.left
                }

                val rightChild = node.right ?: return node
                val minNode = findMin(rightChild)
                node.key = minNode.key
                node.value = minNode.value
                node.right = deleteNode(node.right, minNode.key)
            }
        }
        return node
    }

    private fun findMin(node: BinNode<K, V>): BinNode<K, V> {
        var current = node
        while (true) {
            val leftChild = current.left ?: break
            current = leftChild
        }
        return current
    }

    override fun search(key: K): V? {
        var current = root

        while (current != null) {
            when {
                key < current.key -> current = current.left
                key > current.key -> current = current.right
                else -> return current.value
            }
        }

        return null
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<BinNode<K, V>>()
        root?.let { queue.add(it) }

        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }

    var value : V,
    var left: N? = null,
    var right: N? = null

class RBNode<K: Comparable<K>, V>(
    key: K,
    value: V,
    left: RBNode<K, V>? = null,
    right: RBNode<K, V>? = null,
    var parent: RBNode<K, V>? = null,
    var color: NodeColor = NodeColor.RED,
): Node<K, V, RBNode<K, V>>(key, value, left, right)

class RBTree<K: Comparable<K>, V>: Tree<K, V> {
    private var root: RBNode<K, V>? = null

    override fun insert(key: K, value: V) {
        val newNode = RBNode(key, value)

        if (root == null) {
            root = newNode
        } else {
            insertBST(root, newNode)
        }

        fixAfterInsert(newNode)
    }

    override fun delete(key: K) {
        val node = findNode(key) ?: return
        deleteNode(node)
    }

    override fun search(key: K): V? {
        val node = findNode(key)
        return node?.value
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun insertBST(current: RBNode<K, V>?, node: RBNode<K, V>) {
        if (current != null) {
            when {
                node.key < current.key -> {
                    val currentLeftChild = current.left

                    if (currentLeftChild == null) {
                        current.left = node
                        node.parent = current
                    } else {
                        insertBST(currentLeftChild, node)
                    }
                }

                node.key > current.key -> {
                    val currentRightChild = current.right

                    if (currentRightChild == null) {
                        current.right = node
                        node.parent = current
                    } else {
                        insertBST(currentRightChild, node)
                    }
                }

                else -> {
                    current.value = node.value
                }
            }
        }
    }

    private fun deleteNode(node: RBNode<K, V>?) {
        var y = node
        var yOriginalColor = y?.color
        var x: RBNode<K, V>?

        if (node?.left == null) {
            x = node?.right
            transplant(node, node?.right)
        } else if (node.right == null) {                        
            x = node.left
            transplant(node, node.left)
        } else {
            y = minimum(node.right)
            yOriginalColor = y?.color
            x = y?.right

            if (y?.parent == node) {
                x?.parent = y
            } else {
                transplant(y, y?.right)
                y?.right = node.right
                y?.right?.parent = y
            }

            transplant(node, y)
            y?.left = node.left
            y?.left?.parent = y
            y?.color = node.color
        }

        if (yOriginalColor == NodeColor.BLACK) {
            fixAfterDelete(x)
        }
    }

    private fun findNode(key: K): RBNode<K, V>? {
        var current = root

        while (current != null) {
            current = when {
                key > current.key -> current.right
                key < current.key -> current.left
                else -> return current
            }
        }

        return null
    }

    private fun rotateLeft(x: RBNode<K, V>?) {         //x        y
        val y = x?.right ?: return                     // \  ->  /
                                                       //  y    x
        x.right = y.left
        y.left?.parent = x

        val parent = x.parent
        y.parent = parent

        when {
            parent == null -> root = y
            x == parent.left -> parent.left = y
            else -> parent.right = y
        }

        y.left = x
        x.parent = y
    }

    private fun rotateRight(x: RBNode<K, V>?) {        // x     y
        val y = x?.left ?: return                     // /  ->   \
                                                     // y         x
        x.left = y.right
        y.right?.parent = x

        val parent = x.parent
        y.parent = parent

        when {
            parent == null -> root = y
            x == parent.right -> parent.right = y
            else -> parent.left = y
        }

        y.right = x
        x.parent = y
    }

    private fun fixAfterInsert(node: RBNode<K, V>) {
        var current: RBNode<K, V>? = node

        while (current != root && current?.parent?.color == NodeColor.RED) {
            val grandparent = current.parent?.parent

            if (current.parent == grandparent?.left) {
                val uncle = grandparent?.right

                if (uncle?.color == NodeColor.RED) {
                    current.parent?.color = NodeColor.BLACK
                    uncle.color = NodeColor.BLACK
                    grandparent.color = NodeColor.RED
                    current = grandparent
                } else {
                    if (current == current.parent?.right) {
                        val parent = current.parent

                        if (parent != null) {
                            current = parent
                            rotateLeft(current)
                        }
                    }

                    current.parent?.color = NodeColor.BLACK
                    grandparent?.color = NodeColor.RED
                    rotateRight(grandparent)
                }
            } else {
                val uncle = grandparent?.left

                if (uncle?.color == NodeColor.RED) {
                    current.parent?.color = NodeColor.BLACK
                    uncle.color = NodeColor.BLACK
                    grandparent.color = NodeColor.RED
                    current = grandparent
                } else {
                    if (current == current.parent?.left) {
                        current = current.parent
                        rotateRight(current)
                    }

                    current?.parent?.color = NodeColor.BLACK
                    grandparent?.color = NodeColor.RED
                    rotateLeft(grandparent)
                }
            }
        }

        root?.color = NodeColor.BLACK
    }

    private fun minimum(node: RBNode<K, V>?): RBNode<K, V>? {
        var current = node

        while (current?.left != null) {
            current = current.left
        }

        return current
    }

    private fun transplant(x: RBNode<K, V>?, y: RBNode<K, V>?) {
        when {
            x?.parent == null -> root = y                   //   P
            x == x.parent?.left -> x.parent?.left = y       //  /            P
            else -> x.parent?.right = y                     // x      ->    /
        }                                                  // / \          y
                                                          // L   R
        y?.parent = x?.parent
    }

    private fun fixAfterDelete(node: RBNode<K, V>?) {
        var current = node

        while (current != root && current?.color != NodeColor.BLACK) {
            val parent = current?.parent ?: break

            if (current == parent.left) {
                var brother = parent.right

                if (brother?.color == NodeColor.RED) {
                    brother.color = NodeColor.BLACK
                    parent.color = NodeColor.RED
                    rotateLeft(parent)
                    brother = parent.right
                } else {
                    if (brother?.left?.color == NodeColor.BLACK && brother.right?.color == NodeColor.BLACK) {
                        brother.color = NodeColor.RED
                        current = parent
                    } else {
                        if (brother?.right?.color == NodeColor.BLACK) {
                            brother.left?.color = NodeColor.BLACK
                            brother.color = NodeColor.RED
                            rotateRight(brother)
                            brother = parent.right
                        }

                        brother?.color = parent.color
                        parent.color = NodeColor.BLACK
                        brother?.right?.color = NodeColor.BLACK
                        rotateLeft(parent)
                        current = root
                    }
                }
            } else {
                var brother = parent.left

                if (brother?.color == NodeColor.RED) {
                    brother.color = NodeColor.BLACK
                    parent.color = NodeColor.RED
                    rotateRight(parent)
                    brother = parent.left
                } else {
                    if (brother?.left?.color == NodeColor.BLACK && brother.right?.color == NodeColor.BLACK) {
                        brother.color = NodeColor.RED
                        current = parent
                    } else {
                        if (brother?.left?.color == NodeColor.BLACK) {
                            brother.right?.color = NodeColor.BLACK
                            brother.color = NodeColor.RED
                            rotateLeft(brother)
                            brother = parent.left
                        }

                        brother?.color = parent.color
                        parent.color = NodeColor.BLACK
                        brother?.left?.color = NodeColor.BLACK
                        rotateRight(parent)
                        current = root
                    }
                }
            }
        }

        current?.color = NodeColor.BLACK
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<RBNode<K, V>>()
        root?.let { queue.add(it) }

        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }
}

        assertEquals("B", tree.search(10))
    }

    fun `bfs iterator returns elements`() {
        val tree = RBTree<Int, String>()

        tree.insert(10, "A")
        tree.insert(5, "B")
        tree.insert(20, "C")

        val result = tree.iterator().asSequence().toList()

        assertEquals(3, result.size)
        assertTrue(result.contains(10 to "A"))
        assertTrue(result.contains(5 to "B"))
        assertTrue(result.contains(20 to "C"))
    }

        }

        for (i in 1..100) {
            tree.delete(i)
        }

        for (i in 1..100) {
            assertNull(tree.search(i))
        }
    }

    @Test
    fun `random operations`() {
        val tree = RBTree<Int, Int>()
        val map = mutableMapOf<Int, Int>()

        val keys = (1..100).shuffled()

        for (k in keys) {
            tree.insert(k, k)
            map[k] = k
        }

        for (k in keys.shuffled()) {
            assertEquals(map[k], tree.search(k))
            tree.delete(k)
            map.remove(k)
        }

        for (k in keys) {
            assertNull(tree.search(k))
        }
    }

    fun `random insert search`() {

            val tree = RBTree<Int, Int>()
            val set = mutableSetOf<Int>()

            val size = Random.nextInt(1, 100)

            repeat(size) {
                val key = Random.nextInt(0, 1000)
                tree.insert(key, key)
                set.add(key)
            }

            for (k in set) {
                assertEquals(k, tree.search(k))
            }
        }
    }

    @Test
    fun `random insert delete`() {
        repeat(500) {
            val tree = RBTree<Int, Int>()
            val set = mutableSetOf<Int>()

            val keys = List(Random.nextInt(1, 100)) {
                Random.nextInt(0, 1000)
            }

            for (k in keys) {
                tree.insert(k, k)
                set.add(k)
            }

            for (k in keys.shuffled()) {
                tree.delete(k)
                set.remove(k)

                for (x in set) {
                    assertEquals(x, tree.search(x))
                }
            }

            for (k in keys) {
                assertNull(tree.search(k))
            }
        }
    }

ssize_t read_message(FILE *stream, void *buf) {
    uint8_t *output = buf;
    size_t bytes_read = 0;

    int reading = 1; //to know when to stop
    uint8_t current_byte = 0;
    int bit_count = 0; //bits in current byte
    int consecutive_ones = 0; //to track when there's 5 times 1 in a row

    int searching = 1;
    uint8_t marker_buffer = 0;
    int marker_bits = 0;
    int leftover_bit_pos = -1;  // Track where we stopped in the current byte
    int leftover_byte = 0;
    int zeros_count = 0;
    int all_bits = 0;

    while (searching) {
        int c = getc(stream);
        if (c == EOF) {
            fprintf(stderr, "Error: EOF while searching for start marker\n");
            return -1;
        }
        for (int bit_pos = 0; bit_pos < 8; bit_pos++) {
            int bit = (c >> (7 - bit_pos)) & 1;
            
            marker_buffer = (marker_buffer << 1) | bit;
            marker_bits++;
            if (bit == 0) {
                zeros_count++;
            }
            
            if (marker_bits > 8) {
                marker_bits = 8;
                marker_buffer &= 0xFF;
            }
            
            if (marker_bits == 8 && marker_buffer == 0x7E) {
                if (zeros_count == 2) {
                    searching = 0;
                    leftover_byte = c;
                    leftover_bit_pos = bit_pos + 1;
                    break;
                }
                else {
                    fprintf(stderr, "Error: 0s before start marker");
                    return -1;
                }
            }
        }
    }

    zeros_count = 0;

    if (leftover_bit_pos != -1 && leftover_bit_pos < 8) {
        for (int bit_pos = leftover_bit_pos; bit_pos < 8; bit_pos++) {
            int bit = (leftover_byte >> (7 - bit_pos)) & 1;
            
            if (consecutive_ones == 5) {
                if (bit == 0) {
                    consecutive_ones = 0;
                    zeros_count++;
                    continue;
                } else {
                    fprintf(stderr, "Error: Empty frame or invalid sequence\n");
                    return -1;
                }
            }
            
            current_byte = (current_byte << 1) | bit;
            bit_count++;
            all_bits++;
            
            if (bit == 1) {
                consecutive_ones++;
            } else {
                consecutive_ones = 0;
                zeros_count++;
            }
            
            if (bit_count == 8) {
                if (bytes_read >= MAX_MESSAGE_LEN) {
                    fprintf(stderr, "Error: Message too long\n");
                    return -1;
                }
                output[bytes_read] = current_byte;
                bytes_read++;
                current_byte = 0;
                bit_count = 0;
            }
        }
    }

    while (reading) {
        int c = getc(stream);
        if (c == EOF) {
            fprintf(stderr, "Error: EOF while reading message\n");
            return -1;
        }
        for (int bit_pos = 0; bit_pos < 8; bit_pos++) {
            int bit = (c >> (7 - bit_pos)) & 1;

            if (consecutive_ones == 5) {
                if (bit == 0){
                    zeros_count++;
                    consecutive_ones = 0;                    
                    continue;
                }
            }
        
            if (consecutive_ones == 6) {
                if (bit == 0 && zeros_count != 0) {
                    reading = 0;
                    bit_count = 0;
                    all_bits -= 7;
                    break;
                } else {
                    fprintf (stderr, "Error: Invalid end marker");
                    return -1;
                }
            }
            current_byte = (current_byte << 1) | bit;
            bit_count++;
            all_bits++;

            if (bit == 1) {
                consecutive_ones++;
            }
            else {
                consecutive_ones = 0;
                zeros_count++;
            }

            if (bit_count == 8) {
                if (bytes_read >= MAX_MESSAGE_LEN) {
                    fprintf(stderr, "Error: Message too long\n");
                    return -1;
                }
                output[bytes_read] = current_byte;
                bytes_read++;
                current_byte = 0;
                bit_count = 0;
            }
        }

    }

    if (zeros_count == 0) {
        fprintf(stderr, "Error: Invalid sequence or end marker");
        return -1;
    }

    if (all_bits % 8 != 0){
        fprintf(stderr, "Error: Message contains incomplete byte\n");
        return -1;
    }

    if (bit_count != 0) {
        fprintf(stderr, "Error: Message contains incomplete byte\n");
        return -1;
    }

    return bytes_read;
}

ssize_t write_message(FILE* stream, const void *buf, size_t nbyte) {

    if (buf == NULL && nbyte > 0) {
        fprintf(stderr, "Error: NULL buffer with non-zero size\n");
        return -1;
    }
    const uint8_t *data = buf;
    size_t bytes_written_to_stream = 0;

    uint8_t current_byte = 0;
    int bit_count = 0;
    int consecutive_ones = 0;

    if (putc(0x7E, stream) == EOF) {
        fprintf(stderr, "Error: Failed to write start marker\n");
        return -1;
    }
    bytes_written_to_stream++;
    
    for (size_t i = 0; i < nbyte; i++) {
        uint8_t byte = data[i];
        for (int bit_pos = 0; bit_pos < 8; bit_pos++) {
            int bit = (byte >> (7 - bit_pos)) & 1;
            current_byte = (current_byte << 1) | bit;
            bit_count++;

            if (bit == 1) {
                consecutive_ones++;
            }
            else {
                consecutive_ones = 0;
            }
            if (consecutive_ones == 5) {
                if (bit_count == 8) {
                    if (putc(current_byte, stream) == EOF) {
                        fprintf(stderr, "Error: Failed to write byte\n");
                        return -1;
                    }
                    bytes_written_to_stream++;
                    bit_count = 0;
                    current_byte = 0;
                }

                current_byte = (current_byte << 1) | 0;
                bit_count++;
                consecutive_ones = 0;

                if (bit_count == 8) {
                    if (putc(current_byte, stream) == EOF) {
                        fprintf(stderr, "Error: Failed to write byte\n");
                        return -1;
                    }
                
                bytes_written_to_stream++;
                bit_count = 0;
                current_byte = 0;
                }
            }

            if (bit_count == 8) {
                if (putc(current_byte, stream) == EOF) {
                    fprintf(stderr, "Error: Failed to write byte\n");
                    return -1;
                }
                
                bytes_written_to_stream++;
                bit_count = 0;
                current_byte = 0;
            }
        }
    }
    uint8_t end_marker = 0x7E;
    if (bit_count > 0) {
        for (int bit_pos = 0; bit_pos < 8; bit_pos++) {
            int bit = (end_marker >> (7 - bit_pos)) & 1;
            current_byte = (current_byte << 1) | bit;
            bit_count++;
            
            if (bit_count == 8) {
                if (putc(current_byte, stream) == EOF) return -1;
                bytes_written_to_stream++;
                current_byte = 0;
                bit_count = 0;
            }
        }
        
        if (bit_count > 0) {
            current_byte = current_byte << (8 - bit_count);
            current_byte |= (1 << (8 - bit_count)) - 1;
            if (putc(current_byte, stream) == EOF) return -1;
            bytes_written_to_stream++;
        }
    } 
    else {
        if (putc(end_marker, stream) == EOF) return -1;
        bytes_written_to_stream++;
    }

    return nbyte;
}

/**
 * A mock getc function that get character from buffer
 * @param[in] stream
 * @return reads the next character from buffer and returns it
 * as an unsigned char cast to an int, or EOF when nothing is left
 * in the buffer
 */
int __wrap_getc(FILE *stream) {
    assert_true(stream == message_stream);
    return buffer_getc_index == buffer_getc_len
           ? EOF
           : buffer_getc[buffer_getc_index++];
}

/**
 * A mock putc function that put character to buffer
 * @param[in] c
 * @param[in] stream
 * @return the character written as an unsigned char cast to an int or EOF when buffer is full
 */
int __wrap_putc(int c, FILE *stream) {
    assert_true(stream == message_stream);
    if (buffer_putc_len == TEST_BUFFER_SIZE) {
        return EOF;
    }
    buffer_putc[buffer_putc_len++] = (uint8_t) c;
    return c;
}

    va_start(args, format);
    return_value = vsnprintf(buffer_stderr, sizeof(buffer_stderr),
                             format, args);
    va_end(args);
    return return_value;
}
