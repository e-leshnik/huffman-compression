class AvlTree<K : Comparable<K>, V> : Tree<K, V> {

    }

    private fun updateHeight(node: AvlNode<K, V>?) {
        node?.height = 1 + maxOf(height(node.left), height(node.right))
    }

    private fun balanceFactor(node: AvlNode<K, V>?): Int {
        return if (node == null) 0 else height(node.left) - height(node.right)
    }

    private fun minNode(node: AvlNode<K, V>): AvlNode<K, V> {
        var current: AvlNode<K, V> = node
        while (current.left != null) {
            current = current.left!!
        }
        return current
    }

enum class NodeColor {
    RED, BLACK
}

class RBNode<K : Comparable<K>, V>(
    key: K,
    value: V,
    left: RBNode<K, V>? = null,
    right: RBNode<K, V>? = null,
    var parent: RBNode<K, V>? = null,
    var color: NodeColor = NodeColor.RED,
) : Node<K, V, RBNode<K, V>>(key, value, left, right)

class RBTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: RBNode<K, V>? = null

    private fun rotateLeft(x: RBNode<K, V>?) {         //x        y
        val y = x?.right ?: return                     // \  ->  /
                                                       //  y    x
        x.right = y.left
        y.left?.parent = x

        val parent = x.parent
        y.parent = parent

        when {
            parent == null -> root = y
            x == parent.left -> parent.left = y
            else -> parent.right = y
        }

        y.left = x
        x.parent = y
    }

    private fun rotateRight(x: RBNode<K, V>?) {        // x     y
        val y = x?.left ?: return                     // /  ->   \
                                                     // y         x
        x.left = y.right
        y.right?.parent = x

        val parent = x.parent
        y.parent = parent

        when {
            parent == null -> root = y
            x == parent.right -> parent.right = y
            else -> parent.left = y
        }

        y.right = x
        x.parent = y
    }

    private fun fixAfterInsert(node: RBNode<K, V>) {
        var current: RBNode<K, V>? = node

        while (current != root && current?.parent?.color == NodeColor.RED) {
            val grandparent = current.parent?.parent

            if (current.parent == grandparent?.left) {
                val uncle = grandparent?.right

                if (uncle?.color == NodeColor.RED) {
                    current.parent?.color = NodeColor.BLACK
                    uncle.color = NodeColor.BLACK
                    grandparent.color = NodeColor.RED
                    current = grandparent
                } else {
                    if (current == current.parent?.right) {
                        val parent = current.parent

                        if (parent != null) {
                            current = parent
                            rotateLeft(current)
                        }
                    }

                    current.parent?.color = NodeColor.BLACK
                    grandparent?.color = NodeColor.RED
                    rotateRight(grandparent)
                }
            } else {
                val uncle = grandparent?.left

                if (uncle?.color == NodeColor.RED) {
                    current.parent?.color = NodeColor.BLACK
                    uncle.color = NodeColor.BLACK
                    grandparent.color = NodeColor.RED
                    current = grandparent
                } else {
                    if (current == current.parent?.left) {
                        current = current.parent
                        rotateRight(current)
                    }

                    current?.parent?.color = NodeColor.BLACK
                    grandparent?.color = NodeColor.RED
                    rotateLeft(grandparent)
                }
            }
        }

        root?.color = NodeColor.BLACK
    }

class BinTree<K: Comparable<K>, V>: Tree<K, V> {
    private var root: BinNode<K, V>? = null

    override fun insert(key: K, value: V) {
        if (root == null) {
            root = BinNode(key, value)
            return
        }

        var current = root

        while (current != null) {
            when {
                key < current.key -> {
                    if (current.left == null) {
                        current.left = BinNode(key, value)
                        return
                    }
                    current = current.left
                }

                key > current.key -> {
                    if (current.right == null) {
                        current.right = BinNode(key, value)
                        return
                    }
                    current = current.right
                }

                else -> {
                    current.value = value
                    return
                }
            }
        }
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(binNode: BinNode<K, V>?, key: K): BinNode<K, V>? {
        binNode ?: return null

        when {
            key < binNode.key -> binNode.left = deleteNode(binNode.left, key)
            key > binNode.key -> binNode.right = deleteNode(binNode.right, key)
            else -> {
                if (binNode.left == null && binNode.right == null) {
                    return null
                }
                if (binNode.left == null) {
                    return binNode.right
                }
                if (binNode.right == null) {
                    return binNode.left
                }

                val minNode = findMin(binNode.right!!)
                binNode.key = minNode.key
                binNode.value = minNode.value
                binNode.right = deleteNode(binNode.right, minNode.key)
            }
        }
        return binNode
    }

    private fun findMin(binNode: BinNode<K, V>): BinNode<K, V> {
        var current = binNode
        while (current.left != null) {
            current = current.left!!
        }
        return current
    }

    override fun search(key: K): V? {
        var current = root

        while (current != null) {
            when {
                key < current.key -> current = current.left
                key > current.key -> current = current.right
                else -> return current.value
            }
        }

        return null
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        val result = mutableListOf<Pair<K, V>>()
        val queue = ArrayDeque<BinNode<K, V>>()

        if (root != null) {
            queue.add(root!!)
        }

        while (queue.isNotEmpty()) {
            val current = queue.removeFirst()
            result.add(current.key to current.value)

            if (current.left != null) {
                queue.add(current.left!!)
            }
            if (current.right != null) {
                queue.add(current.right!!)
            }
        }

        return result.iterator()
    }

}class AvlNode<K : Comparable<K>, V>(
    key: K,
    value: V,
    left: AvlNode<K, V>? = null,
    right: AvlNode<K, V>? = null,
    var height: Int = 1
) : Node<K, V, AvlNode<K, V>>(key, value, left, right)

class AvlTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: AvlNode<K, V>? = null
    var size = 0

    private fun height(node: AvlNode<K, V>?): Int {
        return node?.height ?: 0
    }

    private fun updateHeight(node: AvlNode<K, V>?) {
        node?.height = 1 + maxOf(height(node.left), height(node.right))
    }

    private fun balanceFactor(node: AvlNode<K, V>?): Int {
        return if (node == null) 0 else height(node.left) - height(node.right)
    }

    private fun minNode(node: AvlNode<K, V>): AvlNode<K, V> {
        var current: AvlNode<K, V> = node
        while (true) {
            val left = current.left ?: return current
            current = left
        }
    }

    private fun rotateRight(node: AvlNode<K, V>): AvlNode<K, V> {
        val newRoot: AvlNode<K, V> = node.left ?: return node
        val temp = newRoot.right
        newRoot.right = node
        node.left = temp
        updateHeight(node)
        updateHeight(newRoot)
        return newRoot
    }

    private fun rotateLeft(node: AvlNode<K, V>): AvlNode<K, V> {
        val newRoot : AvlNode<K, V> = node.right ?: return node
        val temp = newRoot.left
        newRoot.left = node
        node.right = temp
        updateHeight(node)
        updateHeight(newRoot)
        return newRoot
    }

    private fun rotateRightLeft(node: AvlNode<K, V>): AvlNode<K, V> {
        node.right = node.right?.let { rotateRight(it) }
        return rotateLeft(node)
    }

    private fun rotateLeftRight(node: AvlNode<K, V>): AvlNode<K, V> {
        node.left = node.left?.let { rotateLeft(it) }
        return rotateRight(node)
    }

    private fun balance(node: AvlNode<K, V>): AvlNode<K, V> {
        updateHeight(node)
        val balance = balanceFactor(node)

        if (balance > 1) {
            if (balanceFactor(node.left) >= 0) return rotateRight(node)
            return rotateLeftRight(node)

        }

        if (balance < -1) {
            if (balanceFactor(node.right) <= 0) return rotateLeft(node)
            return rotateRightLeft(node)
        }

        return node
    }

    override fun search(key: K): V? {
        val target = searchNode(root, key)
        if (target != null) return target.value
        return null
    }

    private fun searchNode(node: AvlNode<K, V>?, key: K): AvlNode<K, V>? {
        if (node == null || node.key == key) return node
        if (node.key < key) return searchNode(node.right, key)
        return searchNode(node.left, key)
    }

    override fun insert(key: K, value: V) {
        root = insertNode(root, key, value)
    }

    private fun insertNode(node: AvlNode<K, V>?, key: K, value: V): AvlNode<K, V> {
        if (node == null) {
            size++
            return AvlNode(key, value)
        }
        if (key < node.key) {
            node.left = insertNode(node.left, key, value)
        } else if (key > node.key) {
            node.right = insertNode(node.right, key, value)
        } else {
            node.value = value
            return node
        }
        return balance(node)
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(node: AvlNode<K, V>?, key: K): AvlNode<K, V>? {
        if (node == null) return null
        if (key < node.key) {
            node.left = deleteNode(node.left, key)
        } else if (key > node.key) {
            node.right = deleteNode(node.right, key)
        } else {
            if (node.left == null) {
                size--
                return node.right
            } else if (node.right == null) {
                size--
                return node.left
            }
            node.right?.let { right ->
                val temp = minNode(right)
                node.key = temp.key
                node.value = temp.value
                node.right = deleteNode(node.right, temp.key)
            } ?: error("Right  child should not be null")
        }
        return balance(node)
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<AvlNode<K, V>>()
        root?.let { queue.add(it) }
        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }
}

    value: V,
    left: AvlNode<K, V>? = null,
    right: AvlNode<K, V>? = null,
    var height: Int = 1
) : Node<K, V, AvlNode<K, V>>(key, value, left, right)

class AvlTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: AvlNode<K, V>? = null
    var size = 0

    private fun height(node: AvlNode<K, V>?): Int {
        return node?.height ?: 0
    }

    private fun updateHeight(node: AvlNode<K, V>?) {
        node?.height = 1 + maxOf(height(node.left), height(node.right))
    }

    private fun balanceFactor(node: AvlNode<K, V>?): Int {
        return if (node == null) 0 else height(node.left) - height(node.right)
    }

    private fun minNode(node: AvlNode<K, V>): AvlNode<K, V> {
        var current: AvlNode<K, V> = node
        while (true) {
            val left = current.left ?: return current
            current = left
        }
    }

    private fun rotateRight(node: AvlNode<K, V>): AvlNode<K, V> {
        val newRoot: AvlNode<K, V> = node.left ?: return node
        val temp = newRoot.right
        newRoot.right = node
        node.left = temp
        updateHeight(node)
        updateHeight(newRoot)
        return newRoot
    }

    private fun rotateLeft(node: AvlNode<K, V>): AvlNode<K, V> {
        val newRoot : AvlNode<K, V> = node.right ?: return node
        val temp = newRoot.left
        newRoot.left = node
        node.right = temp
        updateHeight(node)
        updateHeight(newRoot)
        return newRoot
    }

    private fun rotateRightLeft(node: AvlNode<K, V>): AvlNode<K, V> {
        node.right = node.right?.let { rotateRight(it) }
        return rotateLeft(node)
    }

    private fun rotateLeftRight(node: AvlNode<K, V>): AvlNode<K, V> {
        node.left = node.left?.let { rotateLeft(it) }
        return rotateRight(node)
    }

    private fun balance(node: AvlNode<K, V>): AvlNode<K, V> {
        updateHeight(node)
        val balance = balanceFactor(node)

        if (balance > 1) {
            if (balanceFactor(node.left) >= 0) return rotateRight(node)
            return rotateLeftRight(node)

        }

        if (balance < -1) {
            if (balanceFactor(node.right) <= 0) return rotateLeft(node)
            return rotateRightLeft(node)
        }

        return node
    }

    override fun search(key: K): V? {
        val target = searchNode(root, key)
        if (target != null) return target.value
        return null
    }

    private fun searchNode(node: AvlNode<K, V>?, key: K): AvlNode<K, V>? {
        if (node == null || node.key == key) return node
        if (node.key < key) return searchNode(node.right, key)
        return searchNode(node.left, key)
    }

    override fun insert(key: K, value: V) {
        root = insertNode(root, key, value)
    }

    private fun insertNode(node: AvlNode<K, V>?, key: K, value: V): AvlNode<K, V> {
        if (node == null) {
            size++
            return AvlNode(key, value)
        }
        if (key < node.key) {
            node.left = insertNode(node.left, key, value)
        } else if (key > node.key) {
            node.right = insertNode(node.right, key, value)
        } else {
            node.value = value
            return node
        }
        return balance(node)
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(node: AvlNode<K, V>?, key: K): AvlNode<K, V>? {
        if (node == null) return null
        if (key < node.key) {
            node.left = deleteNode(node.left, key)
        } else if (key > node.key) {
            node.right = deleteNode(node.right, key)
        } else {
            if (node.left == null) {
                size--
                return node.right
            } else if (node.right == null) {
                size--
                return node.left
            }
            node.right?.let { right ->
                val temp = minNode(right)
                node.key = temp.key
                node.value = temp.value
                node.right = deleteNode(node.right, temp.key)
            } ?: error("Right  child should not be null")
        }
        return balance(node)
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<AvlNode<K, V>>()
        root?.let { queue.add(it) }
        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }
}
class BinNode<K : Comparable<K>, V>(
    key: K,
    value: V,
    left: BinNode<K, V>? = null,
    right: BinNode<K, V>? = null
) : Node<K, V, BinNode<K, V>>(key, value, left, right)

class BinTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: BinNode<K, V>? = null

    override fun insert(key: K, value: V) {
        if (root == null) {
            root = BinNode(key, value)
            return
        }

        var current = root

        while (current != null) {
            when {
                key < current.key -> {
                    if (current.left == null) {
                        current.left = BinNode(key, value)
                        return
                    }
                    current = current.left
                }

                key > current.key -> {
                    if (current.right == null) {
                        current.right = BinNode(key, value)
                        return
                    }
                    current = current.right
                }

                else -> {
                    current.value = value
                    return
                }
            }
        }
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(node: BinNode<K, V>?, key: K): BinNode<K, V>? {
        node ?: return null

        when {
            key < node.key -> node.left = deleteNode(node.left, key)
            key > node.key -> node.right = deleteNode(node.right, key)
            else -> {
                if (node.left == null && node.right == null) {
                    return null
                }
                if (node.left == null) {
                    return node.right
                }
                if (node.right == null) {
                    return node.left
                }

                val rightChild = node.right ?: return node
                val minNode = findMin(rightChild)
                node.key = minNode.key
                node.value = minNode.value
                node.right = deleteNode(node.right, minNode.key)
            }
        }
        return node
    }

    private fun findMin(node: BinNode<K, V>): BinNode<K, V> {
        var current = node
        while (true) {
            val leftChild = current.left ?: break
            current = leftChild
        }
        return current
    }

    override fun search(key: K): V? {
        var current = root

        while (current != null) {
            when {
                key < current.key -> current = current.left
                key > current.key -> current = current.right
                else -> return current.value
            }
        }

        return null
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<BinNode<K, V>>()
        root?.let { queue.add(it) }

            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }
}class BinNode<K : Comparable<K>, V>(
    key: K,
    value: V,
    left: BinNode<K, V>? = null,
    right: BinNode<K, V>? = null
) : Node<K, V, BinNode<K, V>>(key, value, left, right)

class BinTree<K : Comparable<K>, V> : Tree<K, V> {
    private var root: BinNode<K, V>? = null

    override fun insert(key: K, value: V) {
        if (root == null) {
            root = BinNode(key, value)
            return
        }

        var current = root

        while (current != null) {
            when {
                key < current.key -> {
                    if (current.left == null) {
                        current.left = BinNode(key, value)
                        return
                    }
                    current = current.left
                }

                key > current.key -> {
                    if (current.right == null) {
                        current.right = BinNode(key, value)
                        return
                    }
                    current = current.right
                }

                else -> {
                    current.value = value
                    return
                }
            }
        }
    }

    override fun delete(key: K) {
        root = deleteNode(root, key)
    }

    private fun deleteNode(node: BinNode<K, V>?, key: K): BinNode<K, V>? {
        node ?: return null

        when {
            key < node.key -> node.left = deleteNode(node.left, key)
            key > node.key -> node.right = deleteNode(node.right, key)
            else -> {
                if (node.left == null && node.right == null) {
                    return null
                }
                if (node.left == null) {
                    return node.right
                }
                if (node.right == null) {
                    return node.left
                }

                val rightChild = node.right ?: return node
                val minNode = findMin(rightChild)
                node.key = minNode.key
                node.value = minNode.value
                node.right = deleteNode(node.right, minNode.key)
            }
        }
        return node
    }

    private fun findMin(node: BinNode<K, V>): BinNode<K, V> {
        var current = node
        while (true) {
            val leftChild = current.left ?: break
            current = leftChild
        }
        return current
    }

    override fun search(key: K): V? {
        var current = root

        while (current != null) {
            when {
                key < current.key -> current = current.left
                key > current.key -> current = current.right
                else -> return current.value
            }
        }

        return null
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<BinNode<K, V>>()
        root?.let { queue.add(it) }

        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }

    var value : V,
    var left: N? = null,
    var right: N? = null

class RBNode<K: Comparable<K>, V>(
    key: K,
    value: V,
    left: RBNode<K, V>? = null,
    right: RBNode<K, V>? = null,
    var parent: RBNode<K, V>? = null,
    var color: NodeColor = NodeColor.RED,
): Node<K, V, RBNode<K, V>>(key, value, left, right)

class RBTree<K: Comparable<K>, V>: Tree<K, V> {
    private var root: RBNode<K, V>? = null

    override fun insert(key: K, value: V) {
        val newNode = RBNode(key, value)

        if (root == null) {
            root = newNode
        } else {
            insertBST(root, newNode)
        }

        fixAfterInsert(newNode)
    }

    override fun delete(key: K) {
        val node = findNode(key) ?: return
        deleteNode(node)
    }

    override fun search(key: K): V? {
        val node = findNode(key)
        return node?.value
    }

    override fun iterator(): Iterator<Pair<K, V>> {
        return bfsSequence().iterator()
    }

    private fun insertBST(current: RBNode<K, V>?, node: RBNode<K, V>) {
        if (current != null) {
            when {
                node.key < current.key -> {
                    val currentLeftChild = current.left

                    if (currentLeftChild == null) {
                        current.left = node
                        node.parent = current
                    } else {
                        insertBST(currentLeftChild, node)
                    }
                }

                node.key > current.key -> {
                    val currentRightChild = current.right

                    if (currentRightChild == null) {
                        current.right = node
                        node.parent = current
                    } else {
                        insertBST(currentRightChild, node)
                    }
                }

                else -> {
                    current.value = node.value
                }
            }
        }
    }

    private fun deleteNode(node: RBNode<K, V>?) {
        var y = node
        var yOriginalColor = y?.color
        var x: RBNode<K, V>?

        if (node?.left == null) {
            x = node?.right
            transplant(node, node?.right)
        } else if (node.right == null) {                        
            x = node.left
            transplant(node, node.left)
        } else {
            y = minimum(node.right)
            yOriginalColor = y?.color
            x = y?.right

            if (y?.parent == node) {
                x?.parent = y
            } else {
                transplant(y, y?.right)
                y?.right = node.right
                y?.right?.parent = y
            }

            transplant(node, y)
            y?.left = node.left
            y?.left?.parent = y
            y?.color = node.color
        }

        if (yOriginalColor == NodeColor.BLACK) {
            fixAfterDelete(x)
        }
    }

    private fun findNode(key: K): RBNode<K, V>? {
        var current = root

        while (current != null) {
            current = when {
                key > current.key -> current.right
                key < current.key -> current.left
                else -> return current
            }
        }

        return null
    }

    private fun rotateLeft(x: RBNode<K, V>?) {         //x        y
        val y = x?.right ?: return                     // \  ->  /
                                                       //  y    x
        x.right = y.left
        y.left?.parent = x

        val parent = x.parent
        y.parent = parent

        when {
            parent == null -> root = y
            x == parent.left -> parent.left = y
            else -> parent.right = y
        }

        y.left = x
        x.parent = y
    }

    private fun rotateRight(x: RBNode<K, V>?) {        // x     y
        val y = x?.left ?: return                     // /  ->   \
                                                     // y         x
        x.left = y.right
        y.right?.parent = x

        val parent = x.parent
        y.parent = parent

        when {
            parent == null -> root = y
            x == parent.right -> parent.right = y
            else -> parent.left = y
        }

        y.right = x
        x.parent = y
    }

    private fun fixAfterInsert(node: RBNode<K, V>) {
        var current: RBNode<K, V>? = node

        while (current != root && current?.parent?.color == NodeColor.RED) {
            val grandparent = current.parent?.parent

            if (current.parent == grandparent?.left) {
                val uncle = grandparent?.right

                if (uncle?.color == NodeColor.RED) {
                    current.parent?.color = NodeColor.BLACK
                    uncle.color = NodeColor.BLACK
                    grandparent.color = NodeColor.RED
                    current = grandparent
                } else {
                    if (current == current.parent?.right) {
                        val parent = current.parent

                        if (parent != null) {
                            current = parent
                            rotateLeft(current)
                        }
                    }

                    current.parent?.color = NodeColor.BLACK
                    grandparent?.color = NodeColor.RED
                    rotateRight(grandparent)
                }
            } else {
                val uncle = grandparent?.left

                if (uncle?.color == NodeColor.RED) {
                    current.parent?.color = NodeColor.BLACK
                    uncle.color = NodeColor.BLACK
                    grandparent.color = NodeColor.RED
                    current = grandparent
                } else {
                    if (current == current.parent?.left) {
                        current = current.parent
                        rotateRight(current)
                    }

                    current?.parent?.color = NodeColor.BLACK
                    grandparent?.color = NodeColor.RED
                    rotateLeft(grandparent)
                }
            }
        }

        root?.color = NodeColor.BLACK
    }

    private fun minimum(node: RBNode<K, V>?): RBNode<K, V>? {
        var current = node

        while (current?.left != null) {
            current = current.left
        }

        return current
    }

    private fun transplant(x: RBNode<K, V>?, y: RBNode<K, V>?) {
        when {
            x?.parent == null -> root = y                   //   P
            x == x.parent?.left -> x.parent?.left = y       //  /            P
            else -> x.parent?.right = y                     // x      ->    /
        }                                                  // / \          y
                                                          // L   R
        y?.parent = x?.parent
    }

    private fun fixAfterDelete(node: RBNode<K, V>?) {
        var current = node

        while (current != root && current?.color != NodeColor.BLACK) {
            val parent = current?.parent ?: break

            if (current == parent.left) {
                var brother = parent.right

                if (brother?.color == NodeColor.RED) {
                    brother.color = NodeColor.BLACK
                    parent.color = NodeColor.RED
                    rotateLeft(parent)
                    brother = parent.right
                } else {
                    if (brother?.left?.color == NodeColor.BLACK && brother.right?.color == NodeColor.BLACK) {
                        brother.color = NodeColor.RED
                        current = parent
                    } else {
                        if (brother?.right?.color == NodeColor.BLACK) {
                            brother.left?.color = NodeColor.BLACK
                            brother.color = NodeColor.RED
                            rotateRight(brother)
                            brother = parent.right
                        }

                        brother?.color = parent.color
                        parent.color = NodeColor.BLACK
                        brother?.right?.color = NodeColor.BLACK
                        rotateLeft(parent)
                        current = root
                    }
                }
            } else {
                var brother = parent.left

                if (brother?.color == NodeColor.RED) {
                    brother.color = NodeColor.BLACK
                    parent.color = NodeColor.RED
                    rotateRight(parent)
                    brother = parent.left
                } else {
                    if (brother?.left?.color == NodeColor.BLACK && brother.right?.color == NodeColor.BLACK) {
                        brother.color = NodeColor.RED
                        current = parent
                    } else {
                        if (brother?.left?.color == NodeColor.BLACK) {
                            brother.right?.color = NodeColor.BLACK
                            brother.color = NodeColor.RED
                            rotateLeft(brother)
                            brother = parent.left
                        }

                        brother?.color = parent.color
                        parent.color = NodeColor.BLACK
                        brother?.left?.color = NodeColor.BLACK
                        rotateRight(parent)
                        current = root
                    }
                }
            }
        }

        current?.color = NodeColor.BLACK
    }

    private fun bfsSequence(): Sequence<Pair<K, V>> = sequence {
        val queue = ArrayDeque<RBNode<K, V>>()
        root?.let { queue.add(it) }

        while (queue.isNotEmpty()) {
            val node = queue.removeFirst()
            yield(node.key to node.value)
            node.left?.let { queue.add(it) }
            node.right?.let { queue.add(it) }
        }
    }
}

        assertEquals("B", tree.search(10))
    }

    fun `bfs iterator returns elements`() {
        val tree = RBTree<Int, String>()

        tree.insert(10, "A")
        tree.insert(5, "B")
        tree.insert(20, "C")

        val result = tree.iterator().asSequence().toList()

        assertEquals(3, result.size)
        assertTrue(result.contains(10 to "A"))
        assertTrue(result.contains(5 to "B"))
        assertTrue(result.contains(20 to "C"))
    }

        }

        for (i in 1..100) {
            tree.delete(i)
        }

        for (i in 1..100) {
            assertNull(tree.search(i))
        }
    }

    @Test
    fun `random operations`() {
        val tree = RBTree<Int, Int>()
        val map = mutableMapOf<Int, Int>()

        val keys = (1..100).shuffled()

        for (k in keys) {
            tree.insert(k, k)
            map[k] = k
        }

        for (k in keys.shuffled()) {
            assertEquals(map[k], tree.search(k))
            tree.delete(k)
            map.remove(k)
        }

        for (k in keys) {
            assertNull(tree.search(k))
        }
    }

    fun `random insert search`() {

            val tree = RBTree<Int, Int>()
            val set = mutableSetOf<Int>()

            val size = Random.nextInt(1, 100)

            repeat(size) {
                val key = Random.nextInt(0, 1000)
                tree.insert(key, key)
                set.add(key)
            }

            for (k in set) {
                assertEquals(k, tree.search(k))
            }
        }
    }

    @Test
    fun `random insert delete`() {
        repeat(500) {
            val tree = RBTree<Int, Int>()
            val set = mutableSetOf<Int>()

            val keys = List(Random.nextInt(1, 100)) {
                Random.nextInt(0, 1000)
            }

            for (k in keys) {
                tree.insert(k, k)
                set.add(k)
            }

            for (k in keys.shuffled()) {
                tree.delete(k)
                set.remove(k)

                for (x in set) {
                    assertEquals(x, tree.search(x))
                }
            }

            for (k in keys) {
                assertNull(tree.search(k))
            }
        }
    }

ssize_t read_message(FILE *stream, void *buf) {
    uint8_t *output = buf;
    size_t bytes_read = 0;

    int reading = 1; //to know when to stop
    uint8_t current_byte = 0;
    int bit_count = 0; //bits in current byte
    int consecutive_ones = 0; //to track when there's 5 times 1 in a row

    int searching = 1;
    uint8_t marker_buffer = 0;
    int marker_bits = 0;
    int leftover_bit_pos = -1;  // Track where we stopped in the current byte
    int leftover_byte = 0;
    int zeros_count = 0;
    int all_bits = 0;

    while (searching) {
        int c = getc(stream);
        if (c == EOF) {
            fprintf(stderr, "Error: EOF while searching for start marker\n");
            return -1;
        }
        for (int bit_pos = 0; bit_pos < 8; bit_pos++) {
            int bit = (c >> (7 - bit_pos)) & 1;
            
            marker_buffer = (marker_buffer << 1) | bit;
            marker_bits++;
            if (bit == 0) {
                zeros_count++;
            }
            
            if (marker_bits > 8) {
                marker_bits = 8;
                marker_buffer &= 0xFF;
            }
            
            if (marker_bits == 8 && marker_buffer == 0x7E) {
                if (zeros_count == 2) {
                    searching = 0;
                    leftover_byte = c;
                    leftover_bit_pos = bit_pos + 1;
                    break;
                }
                else {
                    fprintf(stderr, "Error: 0s before start marker");
                    return -1;
                }
            }
        }
    }

    zeros_count = 0;

    if (leftover_bit_pos != -1 && leftover_bit_pos < 8) {
        for (int bit_pos = leftover_bit_pos; bit_pos < 8; bit_pos++) {
            int bit = (leftover_byte >> (7 - bit_pos)) & 1;
            
            if (consecutive_ones == 5) {
                if (bit == 0) {
                    consecutive_ones = 0;
                    zeros_count++;
                    continue;
                } else {
                    fprintf(stderr, "Error: Empty frame or invalid sequence\n");
                    return -1;
                }
            }
            
            current_byte = (current_byte << 1) | bit;
            bit_count++;
            all_bits++;
            
            if (bit == 1) {
                consecutive_ones++;
            } else {
                consecutive_ones = 0;
                zeros_count++;
            }
            
            if (bit_count == 8) {
                if (bytes_read >= MAX_MESSAGE_LEN) {
                    fprintf(stderr, "Error: Message too long\n");
                    return -1;
                }
                output[bytes_read] = current_byte;
                bytes_read++;
                current_byte = 0;
                bit_count = 0;
            }
        }
    }

    while (reading) {
        int c = getc(stream);
        if (c == EOF) {
            fprintf(stderr, "Error: EOF while reading message\n");
            return -1;
        }
        for (int bit_pos = 0; bit_pos < 8; bit_pos++) {
            int bit = (c >> (7 - bit_pos)) & 1;

            if (consecutive_ones == 5) {
                if (bit == 0){
                    zeros_count++;
                    consecutive_ones = 0;                    
                    continue;
                }
            }
        
            if (consecutive_ones == 6) {
                if (bit == 0 && zeros_count != 0) {
                    reading = 0;
                    bit_count = 0;
                    all_bits -= 7;
                    break;
                } else {
                    fprintf (stderr, "Error: Invalid end marker");
                    return -1;
                }
            }
            current_byte = (current_byte << 1) | bit;
            bit_count++;
            all_bits++;

            if (bit == 1) {
                consecutive_ones++;
            }
            else {
                consecutive_ones = 0;
                zeros_count++;
            }

            if (bit_count == 8) {
                if (bytes_read >= MAX_MESSAGE_LEN) {
                    fprintf(stderr, "Error: Message too long\n");
                    return -1;
                }
                output[bytes_read] = current_byte;
                bytes_read++;
                current_byte = 0;
                bit_count = 0;
            }
        }

    }

    if (zeros_count == 0) {
        fprintf(stderr, "Error: Invalid sequence or end marker");
        return -1;
    }

    if (all_bits % 8 != 0){
        fprintf(stderr, "Error: Message contains incomplete byte\n");
        return -1;
    }

    if (bit_count != 0) {
        fprintf(stderr, "Error: Message contains incomplete byte\n");
        return -1;
    }

    return bytes_read;
}

ssize_t write_message(FILE* stream, const void *buf, size_t nbyte) {

    if (buf == NULL && nbyte > 0) {
        fprintf(stderr, "Error: NULL buffer with non-zero size\n");
        return -1;
    }
    const uint8_t *data = buf;
    size_t bytes_written_to_stream = 0;

    uint8_t current_byte = 0;
    int bit_count = 0;
    int consecutive_ones = 0;

    if (putc(0x7E, stream) == EOF) {
        fprintf(stderr, "Error: Failed to write start marker\n");
        return -1;
    }
    bytes_written_to_stream++;
    
    for (size_t i = 0; i < nbyte; i++) {
        uint8_t byte = data[i];
        for (int bit_pos = 0; bit_pos < 8; bit_pos++) {
            int bit = (byte >> (7 - bit_pos)) & 1;
            current_byte = (current_byte << 1) | bit;
            bit_count++;

            if (bit == 1) {
                consecutive_ones++;
            }
            else {
                consecutive_ones = 0;
            }
            if (consecutive_ones == 5) {
                if (bit_count == 8) {
                    if (putc(current_byte, stream) == EOF) {
                        fprintf(stderr, "Error: Failed to write byte\n");
                        return -1;
                    }
                    bytes_written_to_stream++;
                    bit_count = 0;
                    current_byte = 0;
                }

                current_byte = (current_byte << 1) | 0;
                bit_count++;
                consecutive_ones = 0;

                if (bit_count == 8) {
                    if (putc(current_byte, stream) == EOF) {
                        fprintf(stderr, "Error: Failed to write byte\n");
                        return -1;
                    }
                
                bytes_written_to_stream++;
                bit_count = 0;
                current_byte = 0;
                }
            }

            if (bit_count == 8) {
                if (putc(current_byte, stream) == EOF) {
                    fprintf(stderr, "Error: Failed to write byte\n");
                    return -1;
                }
                
                bytes_written_to_stream++;
                bit_count = 0;
                current_byte = 0;
            }
        }
    }
    uint8_t end_marker = 0x7E;
    if (bit_count > 0) {
        for (int bit_pos = 0; bit_pos < 8; bit_pos++) {
            int bit = (end_marker >> (7 - bit_pos)) & 1;
            current_byte = (current_byte << 1) | bit;
            bit_count++;
            
            if (bit_count == 8) {
                if (putc(current_byte, stream) == EOF) return -1;
                bytes_written_to_stream++;
                current_byte = 0;
                bit_count = 0;
            }
        }
        
        if (bit_count > 0) {
            current_byte = current_byte << (8 - bit_count);
            current_byte |= (1 << (8 - bit_count)) - 1;
            if (putc(current_byte, stream) == EOF) return -1;
            bytes_written_to_stream++;
        }
    } 
    else {
        if (putc(end_marker, stream) == EOF) return -1;
        bytes_written_to_stream++;
    }

    return nbyte;
}

/**
 * A mock getc function that get character from buffer
 * @param[in] stream
 * @return reads the next character from buffer and returns it
 * as an unsigned char cast to an int, or EOF when nothing is left
 * in the buffer
 */
int __wrap_getc(FILE *stream) {
    assert_true(stream == message_stream);
    return buffer_getc_index == buffer_getc_len
           ? EOF
           : buffer_getc[buffer_getc_index++];
}

/**
 * A mock putc function that put character to buffer
 * @param[in] c
 * @param[in] stream
 * @return the character written as an unsigned char cast to an int or EOF when buffer is full
 */
int __wrap_putc(int c, FILE *stream) {
    assert_true(stream == message_stream);
    if (buffer_putc_len == TEST_BUFFER_SIZE) {
        return EOF;
    }
    buffer_putc[buffer_putc_len++] = (uint8_t) c;
    return c;
}

/**
 * A mock fprintf function that checks the value of strings printed to the
 * standard error stream or output stream.
 */
int __wrap_fprintf(FILE *file, const char *format, ...) {
    int return_value;
    va_list args;
    assert_true(file == stderr);
    va_start(args, format);
    return_value = vsnprintf(buffer_stderr, sizeof(buffer_stderr),
                             format, args);
    va_end(args);
    return return_value;
}

/**
 * Create test stream with tmpfile()
 *
 * Run once before tests.
 * @param state unused (required by cmocka API)
 */
static int test_group_setup(void **state) {
    (void) state;
    message_stream = tmpfile();
    bzero(buffer_putc, TEST_BUFFER_SIZE);
    buffer_putc_len = 0u;
    uint8_t message[] = { // "write_message"
            0x80, // 10000000
    };
    write_message(message_stream, message, sizeof(message));
    if (buffer_putc_len != 3
        || buffer_putc[0] != 0x7E
        || buffer_putc[2] != 0x7E) {
        return 1;
    }
    if (buffer_putc[1] == 0x80) { // 10000000
        mode = MSB_FIRST;
        return 0;
    }
    if (buffer_putc[1] == 0x01) { // 00000001
        mode = LSB_FIRST;
        return 0;
    }
    return 1;
}

/**
 * A mock getc function that get character from buffer
 * @param[in] stream
 * @return reads the next character from buffer and returns it
 * as an unsigned char cast to an int, or EOF when nothing is left
 * in the buffer
 */
int __wrap_getc(FILE *stream) {
    assert_true(stream == message_stream);
    return buffer_getc_index == buffer_getc_len
           ? EOF
           : buffer_getc[buffer_getc_index++];
}

/**
 * A mock putc function that put character to buffer
 * @param[in] c
 * @param[in] stream
 * @return the character written as an unsigned char cast to an int or EOF when buffer is full
 */
int __wrap_putc(int c, FILE *stream) {
    assert_true(stream == message_stream);
    if (buffer_putc_len == TEST_BUFFER_SIZE) {
        return EOF;
    }
    buffer_putc[buffer_putc_len++] = (uint8_t) c;
    return c;
}

/**
 * A mock fprintf function that checks the value of strings printed to the
 * standard error stream or output stream.
 */
int __wrap_fprintf(FILE *file, const char *format, ...) {
    int return_value;
    va_list args;
    assert_true(file == stderr);
    va_start(args, format);
    return_value = vsnprintf(buffer_stderr, sizeof(buffer_stderr),
                             format, args);
    va_end(args);
    return return_value;
}

/*
 * Wrap for exit(int) function
 **/
void __wrap_exit(int status) {
    (void)status;
    return;
}

/**
 * Create test stream with tmpfile()
 *
 * Run once before tests.
 * @param state unused (required by cmocka API)
 */
static int test_group_setup(void **state) {
    (void) state;
    message_stream = tmpfile();
    bzero(buffer_putc, TEST_BUFFER_SIZE);
    buffer_putc_len = 0u;
    uint8_t message[] = { // "write_message"
            0x80, // 10000000
    };
    write_message(message_stream, message, sizeof(message));
    if (buffer_putc_len != 3
        || buffer_putc[0] != 0x7E
        || buffer_putc[2] != 0x7E) {
        return 1;
    }
    if (buffer_putc[1] == 0x80) { // 10000000
        mode = MSB_FIRST;
        return 0;
    }
    if (buffer_putc[1] == 0x01) { // 00000001
        mode = LSB_FIRST;
        return 0;
    }
    return 1;
}

 * Close test stream.

    return 0;
}

static uint32_t calc_row_size(uint32_t width, uint16_t bit_count) {
  uint32_t bits = width * bit_count;
  return ((bits + 31) / 32) * 4;
}

void bmp_free(bmp_image_t *image) {
  if (!image)
    return;
  if (image->data)
    free(image->data);
  free(image);
}

bool bmp_is_valid(const bmp_image_t *image) {
  if (!image)
    return false;
  if (image->bit_count != 8 && image->bit_count != 24)
    return false;
  if (image->width <= 0 || image->height == 0)
    return false;
  if (!image->data)
    return false;
  return true;
}

bmp_image_t *bmp_read(const char *filename) {
  FILE *file = fopen(filename, "rb");
  if (!file) {
    fprintf(stderr, "Error: Cannot open file '%s'\n", filename);
    return NULL;
  }

  bmp_header_t header;
  bmp_info_t info;

  if (fread(&header, sizeof(header), 1, file) != 1 ||
      fread(&info, sizeof(info), 1, file) != 1) {
    fprintf(stderr, "Error: Cannot read BMP headers from '%s'\n", filename);
    fclose(file);
    return NULL;
  }

  if (header.type != 0x4D42) {
    fprintf(stderr, "Error: File '%s' is not a BMP file (wrong signature)\n",
            filename);
    fclose(file);
    return NULL;
  }

  if (info.size != 40) {
    fprintf(
        stderr,
        "Error: Unsupported BMP header size in '%s' (%u bytes, expected 40)\n",
        filename, info.size);
    fclose(file);
    return NULL;
  }

  if (info.compression != 0) {
    fprintf(stderr, "Error: Compression is not supported in '%s'\n", filename);
    fclose(file);
    return NULL;
  }

  if (info.planes != 1) {
    fprintf(stderr,
            "Error: Invalid number of planes in '%s' (%u, expected 1)\n",
            filename, info.planes);
    fclose(file);
    return NULL;
  }

  if (info.bit_count != 8 && info.bit_count != 24) {
    fprintf(stderr,
            "Error: Unsupported bit depth in '%s' (%u bits, only 8 and 24 "
            "supported)\n",
            filename, info.bit_count);
    fclose(file);
    return NULL;
  }

  if (info.width == 0 || info.height == 0) {
    fprintf(stderr, "Error: Invalid image dimensions in '%s' (%dx%d)\n",
            filename, info.width, info.height);
    fclose(file);
    return NULL;
  }

  fseek(file, 0, SEEK_END);
  long file_size = ftell(file);

  if (header.size > (uint32_t)file_size) {
    fprintf(stderr,
            "Error: BMP header in '%s' claims file size %u bytes, but actual "
            "size is %ld bytes\n",
            filename, header.size, file_size);
    fclose(file);
    return NULL;
  }

  fseek(file, sizeof(header) + sizeof(info), SEEK_SET);

  bmp_image_t *image = malloc(sizeof(bmp_image_t));
  if (!image) {
    fprintf(stderr, "Error: Memory allocation failed for image structure\n");
    fclose(file);
    return NULL;
  }

  image->width = info.width;
  image->height = info.height;
  image->bit_count = info.bit_count;
  image->row_size = calc_row_size((uint32_t)image->width, image->bit_count);

  if (image->bit_count == 8) {
    if (fread(image->palette, 4, 256, file) != 256) {
      fprintf(stderr, "Error: Cannot read palette from '%s'\n", filename);
      fclose(file);
      free(image);
      return NULL;
    }
  }

  uint32_t expected_data_size = image->row_size * image->height;

  if (header.offset + expected_data_size > header.size) {
    fprintf(stderr,
            "Error: Image data in '%s' exceeds size claimed in BMP header\n",
            filename);
    fprintf(stderr, "  Offset: %u, Expected data size: %u, Header size: %u\n",
            header.offset, expected_data_size, header.size);
    fclose(file);
    free(image);
    return NULL;
  }

  if (header.offset + expected_data_size > (uint32_t)file_size) {
    fprintf(stderr, "Error: Image data in '%s' exceeds file size\n", filename);
    fprintf(stderr, "  Offset: %u, Expected data size: %u, File size: %ld\n",
            header.offset, expected_data_size, file_size);
    fclose(file);
    free(image);
    return NULL;
  }

  if (header.offset > header.size ||
      header.offset < sizeof(header) + sizeof(info)) {
    fprintf(stderr, "Error: Invalid data offset in '%s' (%u)\n", filename,
            header.offset);
    fclose(file);
    free(image);
    return NULL;
  }

  if (fseek(file, header.offset, SEEK_SET) != 0) {
    fprintf(stderr, "Error: Cannot seek to pixel data in '%s' at offset %u\n",
            filename, header.offset);
    fclose(file);
    free(image);
    return NULL;
  }

  image->data = malloc(expected_data_size);
  if (!image->data) {
    fprintf(stderr,
            "Error: Cannot allocate memory for pixel data in '%s' (%u bytes)\n",
            filename, expected_data_size);
    fclose(file);
    free(image);
    return NULL;
  }

  size_t bytes_read = fread(image->data, 1, expected_data_size, file);
  if (bytes_read != expected_data_size) {
    fprintf(stderr,
            "Error: Cannot read pixel data from '%s' (read %zu of %u bytes)\n",
            filename, bytes_read, expected_data_size);
    fclose(file);
    free(image->data);
    free(image);
    return NULL;
  }

  fclose(file);
  return image;
}

bool bmp_write(const bmp_image_t *image, const char *filename) {
  if (!bmp_is_valid(image)) {
    fprintf(stderr, "Error: Invalid image\n");
    return false;
  }

  FILE *file = fopen(filename, "wb");
  if (!file) {
    fprintf(stderr, "Error: Cannot create file '%s'\n", filename);
    return false;
  }

  bmp_header_t header;
  bmp_info_t info;

  uint32_t palette_size = (image->bit_count == 8) ? 1024 : 0;
  uint32_t data_size = image->row_size * image->height;

  header.type = 0x4D42;
  header.size = sizeof(header) + sizeof(info) + palette_size + data_size;
  header.reserved = 0;
  header.offset = sizeof(header) + sizeof(info) + palette_size;

  info.size = 40;
  info.width = image->width;
  info.height = image->height;
  info.planes = 1;
  info.bit_count = image->bit_count;
  info.compression = 0;
  info.image_size = data_size;
  info.x_ppm = 0;
  info.y_ppm = 0;
  info.colors_used = (image->bit_count == 8) ? 256 : 0;
  info.colors_important = 0;

  fwrite(&header, sizeof(header), 1, file);
  fwrite(&info, sizeof(info), 1, file);

  if (image->bit_count == 8) {
    fwrite(image->palette, 4, 256, file);
  }

  fwrite(image->data, 1, data_size, file);

  fclose(file);
  return true;
}

int main(int argc, char *argv[]) {
  if (argc != 3) {
    fprintf(stderr, "Error: Expected 2 arguments\n");
    return 1;
  }

  bmp_image_t *img1 = bmp_read(argv[1]);
  bmp_image_t *img2 = bmp_read(argv[2]);

  if (!img1 || !img2) {
    if (img1)
      bmp_free(img1);
    if (img2)
      bmp_free(img2);
    return 1;
  }

  if (img1->bit_count != img2->bit_count) {
    fprintf(stderr, "Error: Different bit depths\n");
    bmp_free(img1);
    bmp_free(img2);
    return 1;
  }

  if (img1->width != img2->width || img1->height != img2->height) {
    int32_t abs_height1 = (img1->height < 0) ? -img1->height : img1->height;
    int32_t abs_height2 = (img2->height < 0) ? -img2->height : img2->height;
    int32_t abs_width1 = (img1->width < 0) ? -img1->width : img1->width;
    int32_t abs_width2 = (img2->width < 0) ? -img2->width : img2->width;

    if (abs_width1 != abs_width2 || abs_height1 != abs_height2) {
      fprintf(stderr, "Error: Different dimensions\n");
      bmp_free(img1);
      bmp_free(img2);
      return 1;
    }
  }

  int diff_count = 0;
  int max_diffs = 100;
  bool first_diff = true;

  int32_t abs_height1 = (img1->height < 0) ? -img1->height : img1->height;
  int32_t abs_width1 = (img1->width < 0) ? -img1->width : img1->width;

  for (int32_t y = 0; y < abs_height1; y++) {
    int32_t row_idx1 = (img1->height < 0) ? y : (abs_height1 - 1 - y);
    int32_t row_idx2 = (img2->height < 0) ? y : (abs_height1 - 1 - y);

    uint8_t *row1 = img1->data + row_idx1 * img1->row_size;
    uint8_t *row2 = img2->data + row_idx2 * img2->row_size;

    for (int32_t x = 0; x < abs_width1; x++) {
      bool equal = false;

      if (img1->bit_count == 8) {
        uint8_t idx1 = row1[x];
        uint8_t idx2 = row2[x];

        equal = (img1->palette[idx1][0] == img2->palette[idx2][0] &&
                 img1->palette[idx1][1] == img2->palette[idx2][1] &&
                 img1->palette[idx1][2] == img2->palette[idx2][2]);
      } else { // 24-bit
        uint8_t *px1 = row1 + x * 3;
        uint8_t *px2 = row2 + x * 3;

        equal = (px1[0] == px2[0] && // Blue
                 px1[1] == px2[1] && // Green
                 px1[2] == px2[2]);  // Red
      }

      if (!equal) {
        if (first_diff) {
          fprintf(stderr, "Next pixels are different:\n");
          first_diff = false;
        }
        fprintf(stderr, "x%-6d y%-6d\n", x, y);
        diff_count++;

        if (diff_count >= max_diffs) {
          y = img1->height;
          break;
        }
      }
    }
  }

  bmp_free(img1);
  bmp_free(img2);

  if (diff_count == 0) {
    printf("Images are same\n");
    return 0;
  } else {
    return 2;
  }
}

      image->palette[i][1] = 255 - image->palette[i][1]; // Green
      image->palette[i][2] = 255 - image->palette[i][2]; // Red
    }
  } else if (image->bit_count == 24) {
    for (int32_t y = 0; y < image->height; y++) {
      uint8_t *row = image->data + y * image->row_size;
      for (int32_t x = 0; x < image->width; x++) {
        uint8_t *pixel = row + x * 3;
        pixel[0] = 255 - pixel[0]; // Blue
        pixel[1] = 255 - pixel[1]; // Green
        pixel[2] = 255 - pixel[2]; // Red
      }
    }
  }

  if (!bmp_write(image, argv[2])) {
    bmp_free(image);
    return 1;
  }

  bmp_free(image);
  return 0;

#define BMP_TYPE 0x4D42

BMPImage *bmp_read(const char *filename) {
  FILE *file = fopen(filename, "rb");
  if (!file) {
    fprintf(stderr, "Error: Cannot open file '%s': %s\n", filename,
            strerror(errno));
    return NULL;
  }

  BMPImage *image = malloc(sizeof(BMPImage));
  if (!image) {
    fprintf(stderr, "Error: Memory allocation failed\n");
    fclose(file);
    return NULL;
  }

  if (fread(&image->file_header, sizeof(BMPFileHeader), 1, file) != 1) {
    fprintf(stderr, "Error: Cannot read BMP file header\n");
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->file_header.type != BMP_TYPE) {
    fprintf(stderr, "Error: Not a valid BMP file (wrong signature)\n");
    free(image);
    fclose(file);
    return NULL;
  }

  if (fread(&image->info_header, sizeof(BMPInfoHeader), 1, file) != 1) {
    fprintf(stderr, "Error: Cannot read BMP info header\n");
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.size != 40) {
    fprintf(stderr, "Error: Unsupported BMP format (header size must be 40)\n");
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.bits_per_pixel != 8 &&
      image->info_header.bits_per_pixel != 24) {
    fprintf(stderr,
            "Error: Unsupported bit depth (%d). Only 8-bit and 24-bit are "
            "supported\n",
            image->info_header.bits_per_pixel);
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.width <= 0 || image->info_header.height == 0) {
    fprintf(stderr, "Error: Invalid image dimensions (width=%d, height=%d)\n",
            image->info_header.width, image->info_header.height);
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.compression != 0) {
    fprintf(stderr, "Error: Compression is not supported\n");
    free(image);
    fclose(file);
    return NULL;
  }

  image->color_table = NULL;
  if (image->info_header.bits_per_pixel == 8) {
    uint32_t color_table_size = image->info_header.colors_used;
    if (color_table_size == 0) {
      color_table_size = 256;
    }

    image->color_table = malloc(color_table_size * sizeof(RGBQuad));
    if (!image->color_table) {
      fprintf(stderr, "Error: Memory allocation for color table failed\n");
      free(image);
      fclose(file);
      return NULL;
    }

    size_t read_count =
        fread(image->color_table, sizeof(RGBQuad), color_table_size, file);
    if (read_count != color_table_size) {
      fprintf(stderr, "Error: Cannot read color table (expected %u, got %zu)\n",
              color_table_size, read_count);
      free(image->color_table);
      free(image);
      fclose(file);
      return NULL;
    }
  } else {
    fseek(file, image->file_header.offset, SEEK_SET);
  }

  image->row_size = bmp_calculate_row_size((uint32_t)image->info_header.width,
                                           image->info_header.bits_per_pixel);

  uint32_t height = (uint32_t)abs(image->info_header.height);
  image->actual_image_size = image->row_size * height;

  uint32_t required_size = image->file_header.offset + image->actual_image_size;
  if (required_size > image->file_header.size) {
    fprintf(stderr, "Error: Image data exceeds size claimed in BMP header\n");
    fprintf(stderr, "  Offset: %u, Image data size: %u, Header file size: %u\n",
            image->file_header.offset, image->actual_image_size,
            image->file_header.size);
    if (image->color_table)
      free(image->color_table);
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.image_size != 0) {
    if (image->info_header.image_size > image->actual_image_size) {
      uint32_t required_with_header_size =
          image->file_header.offset + image->info_header.image_size;
      if (required_with_header_size > image->file_header.size) {
        fprintf(stderr, "Error: Image size in header exceeds file size\n");
        if (image->color_table)
          free(image->color_table);
        free(image);
        fclose(file);
        return NULL;
      }
    }
  }

  image->pixel_data = malloc(image->actual_image_size);
  if (!image->pixel_data) {
    fprintf(stderr, "Error: Memory allocation for pixel data failed\n");
    if (image->color_table)
      free(image->color_table);
    free(image);
    fclose(file);
    return NULL;
  }

  size_t read_size =
      fread(image->pixel_data, 1, image->actual_image_size, file);
  if (read_size != image->actual_image_size) {
    fprintf(stderr,
            "Error: Cannot read pixel data (expected %u bytes, got %zu)\n",
            image->actual_image_size, read_size);
    free(image->pixel_data);
    if (image->color_table)
      free(image->color_table);
    free(image);
    fclose(file);
    return NULL;
  }

  fclose(file);
  return image;
}

bool bmp_write(BMPImage *image, const char *filename) {
  if (!bmp_is_valid(image)) {
    fprintf(stderr, "Error: Invalid BMP image structure\n");
    return false;
  }

  FILE *file = fopen(filename, "wb");
  if (!file) {
    fprintf(stderr, "Error: Cannot create file '%s': %s\n", filename,
            strerror(errno));
    return false;
  }

  if (fwrite(&image->file_header, sizeof(BMPFileHeader), 1, file) != 1) {
    fprintf(stderr, "Error: Cannot write BMP file header\n");
    fclose(file);
    return false;
  }

  if (fwrite(&image->info_header, sizeof(BMPInfoHeader), 1, file) != 1) {
    fprintf(stderr, "Error: Cannot write BMP info header\n");
    fclose(file);
    return false;
  }

  if (image->info_header.bits_per_pixel == 8 && image->color_table) {
    uint32_t color_table_size = image->info_header.colors_used;
    if (color_table_size == 0) {
      color_table_size = 256;
    }

    size_t written =
        fwrite(image->color_table, sizeof(RGBQuad), color_table_size, file);
    if (written != color_table_size) {
      fprintf(stderr,
              "Error: Cannot write color table (expected %u, wrote %zu)\n",
              color_table_size, written);
      fclose(file);
      return false;
    }
  }

  size_t written = fwrite(image->pixel_data, 1, image->actual_image_size, file);
  if (written != image->actual_image_size) {
    fprintf(stderr,
            "Error: Cannot write pixel data (expected %u bytes, wrote %zu)\n",
            image->actual_image_size, written);
    fclose(file);
    return false;
  }

  long current_size = ftell(file);
  if (current_size < 0) {
    perror("ftell");
    fclose(file);
    return false;
  }

  if ((uint32_t)current_size < image->file_header.size) {
    uint32_t padding = image->file_header.size - (uint32_t)current_size;
    uint8_t zero = 0;
    for (uint32_t i = 0; i < padding; i++) {
      if (fwrite(&zero, 1, 1, file) != 1) {
        fprintf(stderr, "Error: Cannot write padding\n");
        fclose(file);
        return false;
      }
    }
  } else if ((uint32_t)current_size > image->file_header.size) {
    fprintf(stderr, "Error: Written file larger than header claims\n");
    fclose(file);
    return false;
  }

  fclose(file);
  return true;
}

void bmp_free(BMPImage *image) {
  if (image) {
    if (image->color_table) {
      free(image->color_table);
    }
    if (image->pixel_data) {
      free(image->pixel_data);
    }
    free(image);
  }
}

bool bmp_is_valid(BMPImage *image) {
  if (!image)
    return false;
  if (image->file_header.type != BMP_TYPE)
    return false;
  if (image->info_header.size != 40)
    return false;
  if (image->info_header.bits_per_pixel != 8 &&
      image->info_header.bits_per_pixel != 24)
    return false;
  if (image->info_header.width <= 0)
    return false;
  if (image->info_header.height == 0)
    return false;
  if (image->info_header.compression != 0)
    return false;
  if (!image->pixel_data)
    return false;
  if (image->info_header.bits_per_pixel == 8 && !image->color_table)
    return false;

  uint32_t calculated_row_size = bmp_calculate_row_size(
      (uint32_t)image->info_header.width, image->info_header.bits_per_pixel);
  if (image->row_size != calculated_row_size)
    return false;

  uint32_t height = (uint32_t)abs(image->info_header.height);
  uint32_t calculated_image_size = image->row_size * height;
  if (image->actual_image_size != calculated_image_size)
    return false;

  if (image->info_header.image_size != 0 &&
      image->info_header.image_size != calculated_image_size) {
    if (image->info_header.image_size > calculated_image_size) {
      return false;
    }
  }

  return true;
}

uint8_t *bmp_get_pixel_address(BMPImage *image, int32_t x, int32_t y) {
  if (!image || !image->pixel_data)
    return NULL;
  if (x < 0 || x >= image->info_header.width)
    return NULL;

  int32_t height_abs = abs(image->info_header.height);
  if (y < 0 || y >= height_abs)
    return NULL;

  uint32_t memory_row;
  if (image->info_header.height > 0) {
    memory_row = (uint32_t)y;
  } else {
    memory_row = (uint32_t)(height_abs - 1 - y);
  }

  uint32_t pixel_size = image->info_header.bits_per_pixel / 8;
  if (pixel_size == 0)
    pixel_size = 1;

  uint32_t offset = memory_row * image->row_size + (uint32_t)x * pixel_size;

  if (offset >= image->actual_image_size)
    return NULL;

  return image->pixel_data + offset;
}

bool bmp_is_same_format(BMPImage *img1, BMPImage *img2) {
  if (!img1 || !img2)
    return false;

  return (img1->info_header.bits_per_pixel == img2->info_header.bits_per_pixel);
}

uint32_t bmp_calculate_row_size(uint32_t width, uint16_t bpp) {
  uint32_t row_size = (width * bpp + 7) / 8;
  return (row_size + 3) & ~3;
}

  }

  if (fread(&image->info_header, sizeof(BMPInfoHeader), 1, file) != 1) {
    fprintf(stderr, "Error: Cannot read BMP info header\n");
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.size != 40) {
    fprintf(stderr, "Error: Unsupported BMP format (header size must be 40)\n");
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.bits_per_pixel != 8 &&
      image->info_header.bits_per_pixel != 24) {
    fprintf(stderr,
            "Error: Unsupported bit depth (%d). Only 8-bit and 24-bit are "
            "supported\n",
            image->info_header.bits_per_pixel);
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.width <= 0 || image->info_header.height == 0) {
    fprintf(stderr, "Error: Invalid image dimensions (width=%d, height=%d)\n",
            image->info_header.width, image->info_header.height);
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.compression != 0) {
    fprintf(stderr, "Error: Compression is not supported\n");
    free(image);
    fclose(file);
    return NULL;
  }

  image->color_table = NULL;
  if (image->info_header.bits_per_pixel == 8) {
    uint32_t color_table_size = image->info_header.colors_used;
    if (color_table_size == 0) {
      color_table_size = 256;
    }

    image->color_table = malloc(color_table_size * sizeof(RGBQuad));
    if (!image->color_table) {
      fprintf(stderr, "Error: Memory allocation for color table failed\n");
      free(image);
      fclose(file);
      return NULL;
    }

    size_t read_count =
        fread(image->color_table, sizeof(RGBQuad), color_table_size, file);
    if (read_count != color_table_size) {
      fprintf(stderr, "Error: Cannot read color table (expected %u, got %zu)\n",
              color_table_size, read_count);
      free(image->color_table);
      free(image);
      fclose(file);
      return NULL;
    }
  } else {
    fseek(file, image->file_header.offset, SEEK_SET);
  }

  image->row_size = bmp_calculate_row_size((uint32_t)image->info_header.width,
                                           image->info_header.bits_per_pixel);

  uint32_t height = (uint32_t)abs(image->info_header.height);
  image->actual_image_size = image->row_size * height;

  uint32_t required_size = image->file_header.offset + image->actual_image_size;
  if (required_size > image->file_header.size) {
    fprintf(stderr, "Error: Image data exceeds size claimed in BMP header\n");
    fprintf(stderr, "  Offset: %u, Image data size: %u, Header file size: %u\n",
            image->file_header.offset, image->actual_image_size,
            image->file_header.size);
    if (image->color_table)
      free(image->color_table);
    free(image);
    fclose(file);
    return NULL;
  }

  if (image->info_header.image_size != 0) {
    if (image->info_header.image_size > image->actual_image_size) {
      uint32_t required_with_header_size =
          image->file_header.offset + image->info_header.image_size;
      if (required_with_header_size > image->file_header.size) {
        fprintf(stderr, "Error: Image size in header exceeds file size\n");
        if (image->color_table)
          free(image->color_table);
        free(image);
        fclose(file);
        return NULL;
      }
    }
  }

  image->pixel_data = malloc(image->actual_image_size);
  if (!image->pixel_data) {
    fprintf(stderr, "Error: Memory allocation for pixel data failed\n");
    if (image->color_table)
      free(image->color_table);
    free(image);
    fclose(file);
    return NULL;
  }

  size_t read_size =
      fread(image->pixel_data, 1, image->actual_image_size, file);
  if (read_size != image->actual_image_size) {
    fprintf(stderr,
            "Error: Cannot read pixel data (expected %u bytes, got %zu)\n",
            image->actual_image_size, read_size);
    free(image->pixel_data);
    if (image->color_table)
      free(image->color_table);
    free(image);
    fclose(file);
    return NULL;
  }

  fclose(file);
  return image;
}

bool bmp_write(BMPImage *image, const char *filename) {
  if (!bmp_is_valid(image)) {
    fprintf(stderr, "Error: Invalid BMP image structure\n");
    return false;
  }

  FILE *file = fopen(filename, "wb");
  if (!file) {
    fprintf(stderr, "Error: Cannot create file '%s': %s\n", filename,
            strerror(errno));
    return false;
  }

  if (fwrite(&image->file_header, sizeof(BMPFileHeader), 1, file) != 1) {
    fprintf(stderr, "Error: Cannot write BMP file header\n");
    fclose(file);
    return false;
  }

  if (fwrite(&image->info_header, sizeof(BMPInfoHeader), 1, file) != 1) {
    fprintf(stderr, "Error: Cannot write BMP info header\n");
    fclose(file);
    return false;
  }

  if (image->info_header.bits_per_pixel == 8 && image->color_table) {
    uint32_t color_table_size = image->info_header.colors_used;
    if (color_table_size == 0) {
      color_table_size = 256;
    }

    size_t written =
        fwrite(image->color_table, sizeof(RGBQuad), color_table_size, file);
    if (written != color_table_size) {
      fprintf(stderr,
              "Error: Cannot write color table (expected %u, wrote %zu)\n",
              color_table_size, written);
      fclose(file);
      return false;
    }
  }

  size_t written = fwrite(image->pixel_data, 1, image->actual_image_size, file);
  if (written != image->actual_image_size) {
    fprintf(stderr,
            "Error: Cannot write pixel data (expected %u bytes, wrote %zu)\n",
            image->actual_image_size, written);
    fclose(file);
    return false;
  }

  long current_size = ftell(file);
  if (current_size < 0) {
    perror("ftell");
    fclose(file);
    return false;
  }

  if ((uint32_t)current_size < image->file_header.size) {
    uint32_t padding = image->file_header.size - (uint32_t)current_size;
    uint8_t zero = 0;
    for (uint32_t i = 0; i < padding; i++) {
      if (fwrite(&zero, 1, 1, file) != 1) {
        fprintf(stderr, "Error: Cannot write padding\n");
        fclose(file);
        return false;
      }
    }
  } else if ((uint32_t)current_size > image->file_header.size) {
    fprintf(stderr, "Error: Written file larger than header claims\n");
    fclose(file);
    return false;
  }

  fclose(file);
  return true;
}

void bmp_free(BMPImage *image) {
  if (image) {
    if (image->color_table) {
      free(image->color_table);
    }
    if (image->pixel_data) {
      free(image->pixel_data);
    }
    free(image);
  }
}

bool bmp_is_valid(BMPImage *image) {
  if (!image)
    return false;
  if (image->file_header.type != BMP_TYPE)
    return false;
  if (image->info_header.size != 40)
    return false;
  if (image->info_header.bits_per_pixel != 8 &&
      image->info_header.bits_per_pixel != 24)
    return false;
  if (image->info_header.width <= 0)
    return false;
  if (image->info_header.height == 0)
    return false;
  if (image->info_header.compression != 0)
    return false;
  if (!image->pixel_data)
    return false;
  if (image->info_header.bits_per_pixel == 8 && !image->color_table)
    return false;

  uint32_t calculated_row_size = bmp_calculate_row_size(
      (uint32_t)image->info_header.width, image->info_header.bits_per_pixel);
  if (image->row_size != calculated_row_size)
    return false;

  uint32_t height = (uint32_t)abs(image->info_header.height);
  uint32_t calculated_image_size = image->row_size * height;
  if (image->actual_image_size != calculated_image_size)
    return false;

  if (image->info_header.image_size != 0 &&
      image->info_header.image_size != calculated_image_size) {
    if (image->info_header.image_size > calculated_image_size) {
      return false;
    }
  }

  return true;
}

uint8_t *bmp_get_pixel_address(BMPImage *image, int32_t x, int32_t y) {
  if (!image || !image->pixel_data)
    return NULL;
  if (x < 0 || x >= image->info_header.width)
    return NULL;

  int32_t height_abs = abs(image->info_header.height);
  if (y < 0 || y >= height_abs)
    return NULL;

  uint32_t memory_row;
  if (image->info_header.height > 0) {
    memory_row = (uint32_t)y;
  } else {
    memory_row = (uint32_t)(height_abs - 1 - y);
  }

  uint32_t pixel_size = image->info_header.bits_per_pixel / 8;
  if (pixel_size == 0)
    pixel_size = 1;

  uint32_t offset = memory_row * image->row_size + (uint32_t)x * pixel_size;

  if (offset >= image->actual_image_size)
    return NULL;

  return image->pixel_data + offset;
}

bool bmp_is_same_format(BMPImage *img1, BMPImage *img2) {
  if (!img1 || !img2)
    return false;

  return (img1->info_header.bits_per_pixel == img2->info_header.bits_per_pixel);
}

uint32_t bmp_calculate_row_size(uint32_t width, uint16_t bpp) {
  uint32_t row_size = (width * bpp + 7) / 8;
  return (row_size + 3) & ~3;
}

bool bmp_compare(BMPImage *img1, BMPImage *img2, uint32_t max_differences,
                 uint32_t *differences_found, int32_t **diff_coords_x,
                 int32_t **diff_coords_y) {
  if (!bmp_is_valid(img1) || !bmp_is_valid(img2)) {
    return false;
  }

  if (img1->info_header.bits_per_pixel != img2->info_header.bits_per_pixel) {
    return false;
  }

  if (img1->info_header.width != img2->info_header.width ||
      abs(img1->info_header.height) != abs(img2->info_header.height)) {
    return false;
  }

  *differences_found = 0;
  *diff_coords_x = malloc(max_differences * sizeof(int32_t));
  *diff_coords_y = malloc(max_differences * sizeof(int32_t));

  if (!*diff_coords_x || !*diff_coords_y) {
    free(*diff_coords_x);
    free(*diff_coords_y);
    *diff_coords_x = NULL;
    *diff_coords_y = NULL;
    return false;
  }

  int32_t width = img1->info_header.width;
  uint32_t height = (uint32_t)abs(img1->info_header.height);

  for (uint32_t y = 0; y < height; y++) {
    for (int32_t x = 0; x < width; x++) {
      uint8_t *pixel1 = bmp_get_pixel_address(img1, x, (int32_t)y);
      uint8_t *pixel2 = bmp_get_pixel_address(img2, x, (int32_t)y);

      if (!pixel1 || !pixel2) {
        continue;
      }

      bool different = false;

      if (img1->info_header.bits_per_pixel == 8) {
        uint8_t idx1 = *pixel1;
        uint8_t idx2 = *pixel2;

        uint32_t colors1 =
            img1->info_header.colors_used ? img1->info_header.colors_used : 256;
        uint32_t colors2 =
            img2->info_header.colors_used ? img2->info_header.colors_used : 256;

        if (idx1 >= colors1 || idx2 >= colors2) {
          different = true;
        } else {
          RGBQuad c1 = img1->color_table[idx1];
          RGBQuad c2 = img2->color_table[idx2];
          if (c1.red != c2.red || c1.green != c2.green || c1.blue != c2.blue) {
            different = true;
          }
        }
      } else {
        if (pixel1[0] != pixel2[0] || pixel1[1] != pixel2[1] ||
            pixel1[2] != pixel2[2]) {
          different = true;
        }
      }

      if (different) {
        int32_t output_y = (int32_t)y;

        (*diff_coords_x)[*differences_found] = x;
        (*diff_coords_y)[*differences_found] = output_y;
        (*differences_found)++;

        if (*differences_found >= max_differences) {
          return true;
        }
      }
    }
  }

  return true;
}

  if (argc != 3) {
    fprintf(stderr, "Error: Expected 2 arguments\n");
    return 1;
  }

  const char *filename1 = argv[1];
  const char *filename2 = argv[2];

  BMPImage *image1 = bmp_read(filename1);
  if (!image1) {
    return 1;
  }

  BMPImage *image2 = bmp_read(filename2);
  if (!image2) {
    bmp_free(image1);
    return 1;
  }

  uint32_t differences_found = 0;
  int32_t *diff_coords_x = NULL;
  int32_t *diff_coords_y = NULL;

  const uint32_t max_differences = 100;

  if (!bmp_compare(image1, image2, max_differences, &differences_found,
                   &diff_coords_x, &diff_coords_y)) {
    fprintf(stderr, "Error: Images have different dimensions or bit depths\n");
    bmp_free(image1);
    bmp_free(image2);
    free(diff_coords_x);
    free(diff_coords_y);
    return 1;
  }

  if (differences_found == 0) {
    printf("Images are same\n");
    bmp_free(image1);
    bmp_free(image2);
    free(diff_coords_x);
    free(diff_coords_y);
    return 0;
  } else {
    fprintf(stderr, "Next pixels are different:\n");
    for (uint32_t i = 0; i < differences_found; i++) {
      fprintf(stderr, "x%-6d y%-6d\n", diff_coords_x[i], diff_coords_y[i]);
    }

    bmp_free(image1);
    bmp_free(image2);
    free(diff_coords_x);
    free(diff_coords_y);
    return 2;
  }
}

    return 1;
  }

  negative->file_header = image->file_header;
  negative->info_header = image->info_header;
  negative->row_size = image->row_size;
  negative->actual_image_size = image->actual_image_size;

  if (image->info_header.bits_per_pixel == 8) {
    uint32_t color_table_size = image->info_header.colors_used;
    if (color_table_size == 0) {

    if (!negative->color_table) {

      free(negative);

    for (uint32_t i = 0; i < color_table_size; i++) {
      negative->color_table[i].blue = 255 - image->color_table[i].blue;
      negative->color_table[i].green = 255 - image->color_table[i].green;
      negative->color_table[i].red = 255 - image->color_table[i].red;
      negative->color_table[i].reserved = 0;
    }
  } else {
    negative->color_table = NULL;
  }

  negative->pixel_data = malloc(negative->actual_image_size);
  if (!negative->pixel_data) {
    fprintf(stderr, "Error: Memory allocation for pixel data failed\n");
    if (negative->color_table)
      free(negative->color_table);
    free(negative);
    bmp_free(image);
    return 1;
  }

  memcpy(negative->pixel_data, image->pixel_data, negative->actual_image_size);

  if (image->info_header.bits_per_pixel == 24) {
    uint32_t height = (uint32_t)abs(image->info_header.height);
    for (uint32_t y = 0; y < height; y++) {

        uint8_t *pixel = bmp_get_pixel_address(negative, x, (int32_t)y);
        if (pixel) {
          pixel[0] = 255 - pixel[0]; // Blue
          pixel[1] = 255 - pixel[1]; // Green
          pixel[2] = 255 - pixel[2]; // Red
        }
      }
    }
  }

  if (!bmp_write(negative, output_filename)) {
    bmp_free(image);
    bmp_free(negative);
    return 1;

  bmp_free(image);
  bmp_free(negative);

  return 0;

#define BASE 1000000000
#define BASE_DIGITS 9

typedef struct DigitNode {
  int chunk;
  struct DigitNode *next;
  struct DigitNode *prev;
} DigitNode;

typedef struct {
  DigitNode *head;
  DigitNode *tail;
  int length;
  bool is_negative;
} BigNumber;

typedef struct StackNode {
  BigNumber *number;
  struct StackNode *next;
} StackNode;

typedef struct {
  StackNode *top;
  int size;
} NumberStack;

DigitNode *create_node(int chunk) {
  if (chunk < 0 || chunk >= BASE) {
    return NULL;
  }

  DigitNode *node = (DigitNode *)malloc(sizeof(DigitNode));
  if (!node)
    return NULL;

  node->chunk = chunk;
  node->next = NULL;
  node->prev = NULL;
  return node;
}

void free_bignum(BigNumber *num) {
  if (!num)
    return;

  DigitNode *current = num->head;
  while (current) {
    DigitNode *next = current->next;
    free(current);
    current = next;
  }
  free(num);
}

BigNumber *create_bignum_from_string(const char *str) {
  BigNumber *num = (BigNumber *)calloc(1, sizeof(BigNumber));
  if (!num)
    return NULL;

  int start = 0;
  if (str[0] == '-') {
    num->is_negative = true;
    start = 1;
  }

  if (str[start] == '\0') {
    free(num);
    return NULL;
  }

  size_t raw_len = strlen(str);
  if (raw_len > (size_t)INT_MAX) {
    free(num);
    return NULL;
  }
  int len = (int)(raw_len);

  for (int i = len; i > start;) {
    int chunk_start = (i - BASE_DIGITS > start) ? i - BASE_DIGITS : start;
    int chunk_len = i - chunk_start;

    char chunk_str[BASE_DIGITS + 1] = {0};
    strncpy(chunk_str, str + chunk_start, chunk_len);
    chunk_str[chunk_len] = '\0';

    char *endptr;
    long val = strtol(chunk_str, &endptr, 10);

    if (endptr == chunk_str || *endptr != '\0' || val < 0 || val >= BASE) {
      free_bignum(num);
      return NULL;
    }
    int chunk = (int)val;

    DigitNode *node = create_node(chunk);
    if (!node) {
      free_bignum(num);
      return NULL;
    }

    if (num->length == 0) {
      num->head = node;
      num->tail = node;
    } else {
      node->prev = num->tail;
      num->tail->next = node;
      num->tail = node;
    }
    num->length++;

    i = chunk_start;
  }

  if (num->length == 0) {
    DigitNode *zero = create_node(0);
    num->head = zero;
    num->tail = zero;
    num->length = 1;
    num->is_negative = false;
  } else {
    while (num->length > 1 && num->tail->chunk == 0) {
      DigitNode *prev = num->tail->prev;
      free(num->tail);
      num->tail = prev;
      num->tail->next = NULL;
      num->length--;
    }
    if (num->length == 1 && num->head->chunk == 0) {
      num->is_negative = false;
    }
  }

  return num;
}

NumberStack *create_stack(void) {
  NumberStack *stack = (NumberStack *)malloc(sizeof(NumberStack));
  if (stack) {
    stack->top = NULL;
    stack->size = 0;
  }
  return stack;
}

void push(NumberStack *stack, BigNumber *num) {
  StackNode *node = (StackNode *)malloc(sizeof(StackNode));
  if (!node)
    return;

  node->number = num;
  node->next = stack->top;
  stack->top = node;
  stack->size++;
}

BigNumber *pop(NumberStack *stack) {
  if (!stack || !stack->top)
    return NULL;

  StackNode *node = stack->top;
  BigNumber *num = node->number;
  stack->top = node->next;
  free(node);
  stack->size--;
  return num;
}

void free_stack(NumberStack *stack) {
  while (stack->top) {
    StackNode *node = stack->top;
    stack->top = node->next;
    free_bignum(node->number);
    free(node);
  }
  free(stack);
}

int compare_abs(BigNumber *a, BigNumber *b) {
  if (a->length != b->length) {
    return a->length > b->length ? 1 : -1;
  }

  DigitNode *node_a = a->tail;
  DigitNode *node_b = b->tail;

  while (node_a && node_b) {
    if (node_a->chunk != node_b->chunk) {
      return node_a->chunk > node_b->chunk ? 1 : -1;
    }
    node_a = node_a->prev;
    node_b = node_b->prev;
  }
  return 0;
}

BigNumber *add_abs(BigNumber *a, BigNumber *b) {
  BigNumber *result = (BigNumber *)calloc(1, sizeof(BigNumber));
  if (!result)
    return NULL;

  DigitNode *node_a = a->head;
  DigitNode *node_b = b->head;
  int carry = 0;

  while (node_a || node_b || carry) {
    int sum = carry;
    if (node_a) {
      sum += node_a->chunk;
      node_a = node_a->next;
    }
    if (node_b) {
      sum += node_b->chunk;
      node_b = node_b->next;
    }

    carry = sum / BASE;
    int chunk = sum % BASE;

    DigitNode *new_node = create_node(chunk);
    if (!new_node) {
      free_bignum(result);
      return NULL;
    }

    if (result->length == 0) {
      result->head = new_node;
      result->tail = new_node;
    } else {
      new_node->prev = result->tail;
      result->tail->next = new_node;
      result->tail = new_node;
    }
    result->length++;
  }

  return result;
}

BigNumber *subtract_abs(BigNumber *a, BigNumber *b) {
  BigNumber *result = (BigNumber *)calloc(1, sizeof(BigNumber));
  if (!result)
    return NULL;

  DigitNode *node_a = a->head;
  DigitNode *node_b = b->head;
  int borrow = 0;

  while (node_a || node_b) {
    int diff = (node_a ? node_a->chunk : 0) - borrow;
    if (node_b) {
      diff -= node_b->chunk;
    }

    if (diff < 0) {
      diff += BASE;
      borrow = 1;
    } else {
      borrow = 0;
    }

    DigitNode *new_node = create_node(diff);
    if (!new_node) {
      free_bignum(result);
      return NULL;
    }

    if (result->length == 0) {
      result->head = new_node;
      result->tail = new_node;
    } else {
      new_node->prev = result->tail;
      result->tail->next = new_node;
      result->tail = new_node;
    }
    result->length++;

    if (node_a)
      node_a = node_a->next;
    if (node_b)
      node_b = node_b->next;
  }

  while (result->tail != result->head && result->tail->chunk == 0) {
    DigitNode *prev = result->tail->prev;
    free(result->tail);
    result->tail = prev;
    result->tail->next = NULL;
    result->length--;
  }

  return result;
}

BigNumber *add(BigNumber *a, BigNumber *b) {
  if (a->is_negative == b->is_negative) {
    BigNumber *result = add_abs(a, b);
    if (result) {
      result->is_negative = a->is_negative;
    }
    return result;
  } else {
    int cmp = compare_abs(a, b);
    if (cmp == 0) {
      return create_bignum_from_string("0");
    } else if (cmp > 0) {
      BigNumber *result = subtract_abs(a, b);
      if (result) {
        result->is_negative = a->is_negative;
      }
      return result;
    } else {
      BigNumber *result = subtract_abs(b, a);
      if (result) {
        result->is_negative = b->is_negative;
      }
      return result;
    }
  }
}

BigNumber *subtract(BigNumber *a, BigNumber *b) {
  BigNumber *neg_b = (BigNumber *)malloc(sizeof(BigNumber));
  if (!neg_b)
    return NULL;

  memcpy(neg_b, b, sizeof(BigNumber));
  neg_b->is_negative = !b->is_negative;

  BigNumber *result = add(a, neg_b);
  free(neg_b);
  return result;
}

BigNumber *multiply(BigNumber *a, BigNumber *b) {
  if ((a->length == 1 && a->head->chunk == 0) ||
      (b->length == 1 && b->head->chunk == 0)) {
    return create_bignum_from_string("0");
  }

  BigNumber *result = create_bignum_from_string("0");
  if (!result)
    return NULL;

  int position = 0;
  DigitNode *node_b = b->head;

  while (node_b) {
    BigNumber *partial = (BigNumber *)calloc(1, sizeof(BigNumber));
    if (!partial) {
      free_bignum(result);
      return NULL;
    }

    DigitNode *node_a = a->head;
    long long carry = 0;

    while (node_a || carry) {
      long long product = carry;
      if (node_a) {
        product += (long long)node_a->chunk * node_b->chunk;
        node_a = node_a->next;
      }

      carry = product / BASE;
      int chunk = (int)(product % BASE);

      DigitNode *new_node = create_node(chunk);
      if (!new_node) {
        free_bignum(partial);
        free_bignum(result);
        return NULL;
      }

      if (partial->length == 0) {
        partial->head = new_node;
        partial->tail = new_node;
      } else {
        new_node->prev = partial->tail;
        partial->tail->next = new_node;
        partial->tail = new_node;
      }
      partial->length++;
    }

    for (int i = 0; i < position; i++) {
      DigitNode *zero_node = create_node(0);
      if (!zero_node) {
        free_bignum(partial);
        free_bignum(result);
        return NULL;
      }
      zero_node->next = partial->head;
      partial->head->prev = zero_node;
      partial->head = zero_node;
      partial->length++;
    }

    BigNumber *new_result = add(result, partial);
    free_bignum(result);
    free_bignum(partial);
    result = new_result;

    if (!result)
      return NULL;

    node_b = node_b->next;
    position++;
  }

  result->is_negative = a->is_negative != b->is_negative;
  return result;
}

bool is_number(const char *token) {
  if (!token || !*token)
    return false;

  int start = 0;
  if (token[0] == '-') {
    if (!token[1])
      return false;
    start = 1;
  }

  for (int i = start; token[i]; i++) {
    if (!isdigit(token[i]))
      return false;
  }

  return true;
}

bool is_operator(const char *token) {
  if (is_number(token)) {
    return false;
  }
  return token[0] == '+' || token[0] == '-' || token[0] == '*';
}

void print_bignum(BigNumber *num) {
  if (!num)
    return;

  if (num->is_negative && (num->length > 1 || num->head->chunk != 0)) {
    printf("-");
  }

  printf("%d", num->tail->chunk);

  DigitNode *current = num->tail->prev;
  while (current) {
    printf("%09d", current->chunk);
    current = current->prev;
  }
  printf("\n");
}

int main(void) {
  char *line = NULL;
  size_t buffer_size = 0;

  if (getline(&line, &buffer_size, stdin) == -1) {
    fprintf(stderr, "Error: Failed to read input\n");
    free(line);
    return 2;
  }

  size_t len = strlen(line);
  if (len > 0 && line[len - 1] == '\n') {
    line[len - 1] = '\0';
    len--;
  }

  if (len == 0) {
    fprintf(stderr, "Error: Empty input\n");
    free(line);
    return 2;
  }

  NumberStack *stack = create_stack();
  if (!stack) {
    fprintf(stderr, "Memory allocation error\n");
    return 2;
  }

  char *token = strtok(line, " ");

  while (token != NULL) {
    if (is_number(token)) {
      BigNumber *num = create_bignum_from_string(token);
      if (!num) {
        fprintf(stderr, "Memory allocation error\n");
        free_stack(stack);
        return 2;
      }
      push(stack, num);
    } else if (is_operator(token)) {
      if (stack->size < 2) {
        fprintf(stderr, "Error: Insufficient operands for operator\n");
        free_stack(stack);
        return 1;
      }

      BigNumber *b = pop(stack);
      BigNumber *a = pop(stack);
      BigNumber *result = NULL;

      switch (token[0]) {
      case '+':
        result = add(a, b);
        break;
      case '-':
        result = subtract(a, b);
        break;
      case '*':
        result = multiply(a, b);
        break;
      }

      free_bignum(a);
      free_bignum(b);

      if (!result) {
        fprintf(stderr, "Memory allocation error\n");
        free_stack(stack);
        return 2;
      }

      push(stack, result);

    } else {
      fprintf(stderr, "Error: Invalid character\n");
      free_stack(stack);
      return 1;
    }

    token = strtok(NULL, " ");
  }

  if (stack->size != 1) {
    if (stack->size == 0) {
      fprintf(stderr, "Error: No result\n");
    } else {
      fprintf(stderr, "Error: Operation symbol is missed\n");
    }
    free_stack(stack);
    return 1;
  }

  BigNumber *result = pop(stack);
  print_bignum(result);
  free_bignum(result);
  free_stack(stack);
  free(line);

  return 0;
}

  DigitNode *node_b = b->head;
  int carry = 0;

  while (node_a || node_b || carry) {
    int sum = carry;
    if (node_a) {
      sum += node_a->chunk;
      node_a = node_a->next;
    }
    if (node_b) {
      sum += node_b->chunk;
      node_b = node_b->next;
    }

    carry = sum / BASE;
    int chunk = sum % BASE;

    DigitNode *new_node = create_node(chunk);
    if (!new_node) {
      free_bignum(result);
      return NULL;
    }

    if (result->length == 0) {
      result->head = new_node;
      result->tail = new_node;
    } else {
      new_node->prev = result->tail;
      result->tail->next = new_node;
      result->tail = new_node;
    }
    result->length++;
  }

  return result;
}

BigNumber *subtract_abs(BigNumber *a, BigNumber *b) {
  BigNumber *result = (BigNumber *)calloc(1, sizeof(BigNumber));
  if (!result)
    return NULL;

  DigitNode *node_a = a->head;
  DigitNode *node_b = b->head;
  int borrow = 0;

  while (node_a || node_b) {
    int diff = (node_a ? node_a->chunk : 0) - borrow;
    if (node_b) {
      diff -= node_b->chunk;
    }

    if (diff < 0) {
      diff += BASE;
      borrow = 1;
    } else {
      borrow = 0;
    }

    DigitNode *new_node = create_node(diff);
    if (!new_node) {
      free_bignum(result);
      return NULL;
    }

    if (result->length == 0) {
      result->head = new_node;
      result->tail = new_node;
    } else {
      new_node->prev = result->tail;
      result->tail->next = new_node;
      result->tail = new_node;
    }
    result->length++;

    if (node_a)
      node_a = node_a->next;
    if (node_b)
      node_b = node_b->next;
  }

  while (result->tail != result->head && result->tail->chunk == 0) {
    DigitNode *prev = result->tail->prev;
    free(result->tail);
    result->tail = prev;
    result->tail->next = NULL;
    result->length--;
  }

  return result;
}

BigNumber *add(BigNumber *a, BigNumber *b) {
  if (a->is_negative == b->is_negative) {
    BigNumber *result = add_abs(a, b);
    if (result) {
      result->is_negative = a->is_negative;
    }
    return result;
  } else {
    int cmp = compare_abs(a, b);
    if (cmp == 0) {
      return create_bignum_from_string("0");
    } else if (cmp > 0) {
      BigNumber *result = subtract_abs(a, b);
      if (result) {
        result->is_negative = a->is_negative;
      }
      return result;
    } else {
      BigNumber *result = subtract_abs(b, a);
      if (result) {
        result->is_negative = b->is_negative;
      }
      return result;
    }
  }
}

BigNumber *subtract(BigNumber *a, BigNumber *b) {
  BigNumber *neg_b = (BigNumber *)malloc(sizeof(BigNumber));
  if (!neg_b)
    return NULL;

  memcpy(neg_b, b, sizeof(BigNumber));
  neg_b->is_negative = !b->is_negative;

  BigNumber *result = add(a, neg_b);
  free(neg_b);
  return result;
}

BigNumber *multiply(BigNumber *a, BigNumber *b) {
  if ((a->length == 1 && a->head->chunk == 0) ||
      (b->length == 1 && b->head->chunk == 0)) {
    return create_bignum_from_string("0");
  }

  BigNumber *result = create_bignum_from_string("0");
  if (!result)
    return NULL;

  int position = 0;
  DigitNode *node_b = b->head;

  while (node_b) {
    BigNumber *partial = (BigNumber *)calloc(1, sizeof(BigNumber));
    if (!partial) {
      free_bignum(result);
      return NULL;
    }

    DigitNode *node_a = a->head;
    long long carry = 0;

    while (node_a || carry) {
      long long product = carry;
      if (node_a) {
        product += (long long)node_a->chunk * node_b->chunk;
        node_a = node_a->next;
      }

      carry = product / BASE;
      int chunk = (int)(product % BASE);

      DigitNode *new_node = create_node(chunk);
      if (!new_node) {
        free_bignum(partial);
        free_bignum(result);
        return NULL;
      }

      if (partial->length == 0) {
        partial->head = new_node;
        partial->tail = new_node;
      } else {
        new_node->prev = partial->tail;
        partial->tail->next = new_node;
        partial->tail = new_node;
      }
      partial->length++;
    }

    for (int i = 0; i < position; i++) {
      DigitNode *zero_node = create_node(0);
      if (!zero_node) {
        free_bignum(partial);
        free_bignum(result);
        return NULL;
      }
      zero_node->next = partial->head;
      partial->head->prev = zero_node;
      partial->head = zero_node;
      partial->length++;
    }

    BigNumber *new_result = add(result, partial);
    free_bignum(result);
    free_bignum(partial);
    result = new_result;

    if (!result)
      return NULL;

    node_b = node_b->next;
    position++;
  }

  result->is_negative = a->is_negative != b->is_negative;
  return result;
}

bool is_number(const char *token) {
  if (!token || !*token)
    return false;

  int start = 0;
  if (token[0] == '-') {
    if (!token[1])
      return false;
    start = 1;
  }

  for (int i = start; token[i]; i++) {
    if (!isdigit(token[i]))
      return false;
  }

  return true;
}

bool is_operator(const char *token) {
  if (is_number(token)) {
    return false;
  }
  return token[0] == '+' || token[0] == '-' || token[0] == '*';
}

void print_bignum(BigNumber *num) {
  if (!num)
    return;

  if (num->is_negative && (num->length > 1 || num->head->chunk != 0)) {
    printf("-");
  }

  printf("%d", num->tail->chunk);

  DigitNode *current = num->tail->prev;
  while (current) {
    printf("%09d", current->chunk);
    current = current->prev;
  }
  printf("\n");
}

int main(void) {
  char *line = NULL;
  size_t buffer_size = 0;

  if (getline(&line, &buffer_size, stdin) == -1) {
    fprintf(stderr, "Error: Failed to read input\n");
    free(line);
    return 2;
  }

  size_t len = strlen(line);
  if (len > 0 && line[len - 1] == '\n') {
    line[len - 1] = '\0';
    len--;
  }

  if (len == 0) {
    fprintf(stderr, "Error: Empty input\n");
    free(line);
    return 2;
  }

  NumberStack *stack = create_stack();
  if (!stack) {
    fprintf(stderr, "Memory allocation error\n");
    return 2;
  }

  char *token = strtok(line, " ");

  while (token != NULL) {
    if (is_number(token)) {
      BigNumber *num = create_bignum_from_string(token);
      if (!num) {
        fprintf(stderr, "Memory allocation error\n");
        free_stack(stack);
        return 2;
      }
      push(stack, num);
    } else if (is_operator(token)) {
      if (stack->size < 2) {
        fprintf(stderr, "Error: Insufficient operands for operator\n");
        free_stack(stack);
        return 1;
      }

      BigNumber *b = pop(stack);
      BigNumber *a = pop(stack);
      BigNumber *result = NULL;

      switch (token[0]) {
      case '+':
        result = add(a, b);
        break;
      case '-':
        result = subtract(a, b);
        break;
      case '*':
        result = multiply(a, b);
        break;
      }

      free_bignum(a);
      free_bignum(b);

      if (!result) {
        fprintf(stderr, "Memory allocation error\n");
        free_stack(stack);
        return 2;
      }

      push(stack, result);

    } else {
      fprintf(stderr, "Error: Invalid character\n");
      free_stack(stack);
      return 1;
    }

    token = strtok(NULL, " ");
  }

  if (stack->size != 1) {
    if (stack->size == 0) {
      fprintf(stderr, "Error: No result\n");
    } else {
      fprintf(stderr, "Error: Operation symbol is missed\n");
    }
    free_stack(stack);
    return 1;
  }

  BigNumber *result = pop(stack);
  print_bignum(result);
  free_bignum(result);
  free_stack(stack);
  free(line);

  return 0;
}

        colorScheme = colorScheme,

    )

/*
 * Modelling file for Coverity Scan
 *
 * This is a modeling file for Coverity Scan. Modeling helps to avoid false
 * positives.
 *
 * - A model file can't import any header files.
 * - Therefore only some built-in primitives like int, char and void are
 *   available but not NULL etc.
 * - Modeling doesn't need full structs and typedefs. Rudimentary structs
 *   and similar types are sufficient.
 * - An uninitialized local pointer is not an error. It signifies that the
 *   variable could be either NULL or have some data.
 *
 * Coverity Scan doesn't pick up modifications automatically. The model file
 * must be uploaded by an admin.
 *
 * See also https://scan.coverity.com/models
 */

void leak_memory(void);
void buffer_overflow(void);
void buffer_underflow(void);

void leak_memory(void) {
    int * const temporary = (int*)malloc(sizeof(int));
    *temporary = 0;
}

void buffer_overflow(void) {
    char * const memory = (char*)malloc(sizeof(int));
    memory[sizeof(int)] = '!';
    free(memory);
}

void buffer_underflow(void) {
    char * const memory = (char*)malloc(sizeof(int));
    memory[-1] = '!';
    free(memory);
}

static const char* status_code_strings[] = {
    "Address not found",
    "Connection dropped",
    "Connection timed out",
};

const char* get_status_code_string(const unsigned int status_code) {
    return status_code_strings[status_code];
}

unsigned int string_to_status_code(const char* const status_code_string) {
    unsigned int i;
    for (i = 0; i < sizeof(status_code_strings) /
                    sizeof(status_code_strings[0]); i++) {
        if (strcmp(status_code_strings[i], status_code_string) == 0) {
            return i;
        }
    }
    return ~0U;
}
